33 //! \brief This is the constructor for an object of type ThetaLookUp.
34 //!
35 //! The object contains a lookup table of specified generalized theta function
36 //! values in order to avoid repeated calculations.
37 //! The table is for theta function values in a lattice of size Lx by Ly
38 //! For complex integer arguments Lx + I Ly. There is an option to include
39 //! a complex double values offset to all values.
40 //////////////////////////////////////////////////////////////////////////////
41 utilities::ThetaLookUp::ThetaLookUp(
42  const int Lx, //!< X dimension of the lattice for which
43  //! to generate A table of values Theta (Lx + I Ly)
44  const int Ly, //!< Y dimension of the lattice for which
45  //! to generate A table of values Theta (Lx + I Ly)
46  const double arg1, //!< First argument of generalized theta function
47  const double arg2, //!< Second argument of generalized theta function
48  const dcmplx quasiHoleOffset, //!< A complex double offset to all the Theta
49  //! function arguments in the table i.e we tabulate
50  //! Theta_[arg1,arg2] (x + I y -quasiHoleOffset)
51  const bool restrictSign, //!< Restrict the sign of the arguments to only be
52  //! positive (then map other quadrants onto that)
53  const int increaseDomain) //!< Extend the range of values stored by this factor
54  //! ALSO increase tau by this factor (for special
55  //! application to torus quantum Hall wave functions)
56 {
57  double tau = increaseDomain*(double)Ly/Lx;
58  if(restrictSign) // only generate the positive quadrant (i>0 and j>0)
59  {
60  thetaLookUpTable = new dcmplx[Lx*Ly*increaseDomain*increaseDomain];
61  for(int i=0; i<Lx*increaseDomain; ++i)
62  {
63  for(int j=0; j<Ly*increaseDomain; ++j)
64  {
65  thetaLookUpTable[i*Ly*increaseDomain+j] = thetaFunction::GeneralisedJacobi(
66  arg1, arg2, (dcmplx(i, j)-quasiHoleOffset)/(double)Lx, tau);
67  }
68  }
69  }
70  else // store all possible values (takes 4x the memory)
71  {
72  thetaLookUpTable=new dcmplx[4*Lx*Ly*increaseDomain*increaseDomain];
73  for(int i=0; i<2*Lx*increaseDomain; ++i)
74  {
75  for(int j=0; j<2*Ly*increaseDomain; ++j)
76  {
77  thetaLookUpTable[i*(2*Ly*increaseDomain)+j] = thetaFunction::GeneralisedJacobi(
78  arg1, arg2, (dcmplx(i-Lx, j-Ly)-quasiHoleOffset)/(double)Lx, tau);
79  }
80  }
81  }
82 }

93 //! \brief The generalised Jacobi theta function is given by
94 //! \f$ \sum_{-\infty}^{\infty} exp(-\pi*t*(n+a)^2+2*I*\pi(n+a)(z+b)) \f$.
95 //!
96 //! The following a and b parameters in the generalized theta function
97 //! correspond to the standard definitions of Theta_1,...,Theta_4:
98 //!
99 //! \verbatim
100 //! a b
101 //! Theta_1 0.5 -0.5
102 //! Theta_2 0.5 0.0
103 //! Theta_3 0.0 0.0
104 //! Theta_4 0.0 0.5
105 //!
106 //! \endverbatim
107 //!
108 //! Note: the generalized theta functions are quite badly behaved.
109 //! So it's important to check the convergence of the summations carefully
110 //!
111 //! \return complex LOG of theta function
112 //////////////////////////////////////////////////////////////////////////////
113 dcmplx utilities::thetaFunction::GeneralisedJacobi(
114  const double a, //!< First parameter in generalized theta function
115  const double b, //!< Second parameter in generalized theta function
116  dcmplx z, //!< Complex argument used as written in the description
117  const double t) //!< The 'nome' of the theta function, used as
118  //! written in the description. N.B. sometimes defined as exp(q)
119 {
120  int n;
121  int nbrTerms;
122  dcmplx thetaFunc,prevTheta,nextTerm;
123  dcmplx xShiftFactor=0;
124  dcmplx yShiftFactor=0;
125  int extraTerms = 5+ceil(a)+ceil(b); // set the minimum number of terms
126  // Shift z to be as close as possible to zero and add in the periodic factors
127  // to compensate.
128  while(real(z)>=1.0)
129  {
130  z -= 1.0;
131  xShiftFactor += 2*PI*I*a;
132  }
133  while(real(z)<=-1.0)
134  {
135  z += 1.0;
136  xShiftFactor -= 2*PI*I*a;
137  }
138  while(imag(z)>=t)
139  {
140  z -= I*t;
141  yShiftFactor += (PI*t-2*PI*I*(z+b));
142  }
143  while(imag(z)<=-t)
144  {
145  z += I*t;
146  yShiftFactor -= (PI*t-2*PI*I*(z+b));
147  }
148  // calculate first term in the sum (n=0 term)
149  thetaFunc = exp(PI*a*(-a*t+2.0*I*(z+b)));
150  prevTheta = thetaFunc;
151  // calculate the next pair of terms (n=+1 and n=-1)
152  nextTerm = exp(PI*(a+1.0)*(-(a+1.0)*t+2.0*I*(z+b)))+exp(PI*(a-1.0)*(-(a-1.0)*t+2.0*I*(z+b)));
153  thetaFunc += nextTerm;
154  n = 2;
155  nbrTerms = 2;
156  // Keep going until until adding terms does not change the result
157  while(abs(nextTerm/prevTheta)>thetaTol || extraTerms>0)
158  {
159  prevTheta += nextTerm;
160  // add next pair of terms to the sum
161  nextTerm = exp(PI*(a+n)*(-(a+n)*t+2.0*I*(z+b)));
162  nextTerm += exp(PI*(a-n)*(-(a-n)*t+2.0*I*(z+b)));
163  thetaFunc += nextTerm;
164  ++n;
165  ++nbrTerms;
166  --extraTerms;
167  }
168  thetaFunc = log(thetaFunc)+xShiftFactor+yShiftFactor;
169  // Use high precision if the low precision range overflows
170  if(std::isinf(real(thetaFunc)) || std::isnan(real(thetaFunc)))
171  {
172  // if the double range is overflowed, then we need to use high precision variables
173  // Declare local variables
174  int prec = 256;
175  mpc_t runTot;
176  mpc_t prev,next1,next2;
177  mpc_t compare;
178  mpfr_t absCompare;
179  mpc_init2(runTot, prec);
180  mpc_init2(prev, prec);
181  mpc_init2(next1, prec);
182  mpc_init2(next2, prec);
183  mpc_init2(compare, prec);
184  mpfr_init(absCompare);
185  // calculate first term in the sum (n=0 term)
186  nextTerm = PI*a*(-a*t+2.0*I*(z+b));
187  mpc_set_d_d(runTot, real(nextTerm), imag(nextTerm), MPC_RNDNN);
188  mpc_exp(runTot, runTot, MPC_RNDNN);
189  mpc_set(prev, runTot, MPC_RNDNN);
190  // calculate the next pair of terms (n=+1 and n=-1)
191  nextTerm = PI*(a+1.0)*(-(a+1.0)*t+2.0*I*(z+b));
192  mpc_set_d_d(next1, real(nextTerm), imag(nextTerm), MPC_RNDNN);
193  mpc_exp(next1, next1, MPC_RNDNN);
194  mpc_add(runTot, runTot, next1, MPC_RNDNN);
195  nextTerm = PI*(a-1.0)*(-(a-1.0)*t+2.0*I*(z+b));
196  mpc_set_d_d(next2, real(nextTerm), imag(nextTerm), MPC_RNDNN);
197  mpc_exp(next2, next2, MPC_RNDNN);
198  mpc_add(runTot, runTot, next2, MPC_RNDNN);
199  n = 2;
200  nbrTerms = 2;
201  mpc_sub(compare, runTot, prev, MPC_RNDNN);
202  mpc_div(compare, compare, runTot, MPC_RNDNN);
203  mpc_abs(absCompare, compare, MPFR_RNDN);
204  while(mpfr_get_d(absCompare, MPFR_RNDN)>thetaTol)
205  {
206  mpc_add(prev, prev, next1, MPC_RNDNN);
207  mpc_add(prev, prev, next2, MPC_RNDNN);
208  // add next pair of terms to the sum
209  nextTerm = PI*(a+n)*(-(a+n)*t+2.0*I*(z+b));
210  mpc_set_d_d(next1, real(nextTerm), imag(nextTerm), MPC_RNDNN);
211  mpc_exp(next1, next1, MPC_RNDNN);
212  mpc_add(runTot, runTot, next1, MPC_RNDNN);
213  nextTerm = PI*(a-n)*(-(a-n)*t+2.0*I*(z+b));
214  mpc_set_d_d(next2, real(nextTerm), imag(nextTerm), MPC_RNDNN);
215  mpc_exp(next2, next2, MPC_RNDNN);
216  mpc_add(runTot, runTot, next2, MPC_RNDNN);
217  mpc_sub(compare, runTot, prev, MPC_RNDNN);
218  mpc_div(compare, compare, runTot, MPC_RNDNN);
219  mpc_abs(absCompare, compare, MPFR_RNDN);
220  ++n;
221  ++nbrTerms;
222  }
223  mpc_log(runTot, runTot, MPC_RNDNN);
224  mpc_real(absCompare, runTot, MPFR_RNDN);
225  thetaFunc = mpfr_get_d(absCompare, MPFR_RNDN);
226  mpc_imag(absCompare, runTot, MPFR_RNDN);
227  thetaFunc += dcmplx(0, mpfr_get_d(absCompare, MPFR_RNDN));
228  thetaFunc += xShiftFactor+yShiftFactor;
229  mpc_clear(runTot);
230  mpc_clear(prev);
231  mpc_clear(next1);
232  mpc_clear(next2);
233  mpc_clear(compare);
234  mpfr_clear(absCompare);
235  }
236  return thetaFunc;
237 }
