327  ////////////////////////////////////////////////////////////////////////////////
328  //! \brief Pre calculation of factorial and binomial factors for each term
329  //! occurring the the CF wave function.
330  //!
331  //! Calculates the contents of the facList. This calculation only needs to
332  //! be done in the negative effective case, which requires additional
333  //! factorial factors.
334  ////////////////////////////////////////////////////////////////////////////////
335  void CalcualtePreFactor(
336  const int n, //!< Number of particles
337  const double flux, //!< Flux
338  const int nbrLLs, //!< Number of Landau levels filled
339  utilities::HpWrap<V, P>* facList)//!< An arbitrary precision list of factorial
340  //! ratios to be populated
341  {
342  {
343  utilities::HpWrap<V, P> realTmp;
344  utilities::HpWrap<V, P> *p_list = facList;
345  for(int d=0; d<m_preData->llOccupationNumbers[0]; ++d)
346  {
347  double m = m_preData->cfLz[d];
348  // Calculate normalization of single particle orbitals
349  double norm = FQHE::MonopoleHarmonicNorm(m_preData->effectiveMonopole, 0,
350  m_preData->cfLz[d])*std::pow(FQHE::extraNormFactor, n/2);
351  for(int t=(int)(m_preData->effectiveMonopole+m); t<=(int)(n-1-m_preData->effectiveMonopole+m); ++t, ++p_list)
352  {
353  utilities::DivFactorial<V, P>(n-1-t,n-1-t-m_preData->effectiveMonopole+m, &realTmp);
354  *p_list = realTmp;
355  utilities::DivFactorial<V,P>(t, t-m_preData->effectiveMonopole-m, &realTmp);
356  *p_list *= realTmp;
357  (*p_list).MulBy(m_preData->binomList[(int)(2*m_preData->effectiveMonopole*m_preData->maxBinom+m_preData->effectiveMonopole+m)], (*p_list));
358  (*p_list).MulBy(1.0/norm, *p_list);
359  // This line implements (-1)^t
360  if(t & 1) (*p_list).MulBy(-1.0, *p_list);
361  ////////////////////////////////////////////////////////////////////////////////
362  #if _DEBUG_
363  utilities::cout.DebuggingInfo() << "p_list = " << (*p_list).Get() << std::endl;
364  #endif
365  ////////////////////////////////////////////////////////////////////////////////
366  }
367  }
368  // Now we do the same thing for higher Landau levels
369  for(int ll=1, degen=0, cumulativeDegen=m_preData->llOccupationNumbers[0]; ll<nbrLLs; ++ll, cumulativeDegen+=degen)
370  {
371  degen = m_preData->llOccupationNumbers[ll];
372  for(int d=cumulativeDegen; d<cumulativeDegen+degen; ++d)
373  {
374  double m = m_preData->cfLz[d];
375  // Calculate normalization of single particle orbitals
376  double norm = FQHE::MonopoleHarmonicNorm(m_preData->effectiveMonopole, ll,
377  m_preData->cfLz[d])*std::pow(FQHE::extraNormFactor, n/2);
378  for(int s=0; s<=ll; ++s)
379  {
380  // skip if we are multiplying by a 0 binomial coefficient
381  if(m_preData->effectiveMonopole+m+s>=0 && m_preData->effectiveMonopole+m+s<=2*m_preData->effectiveMonopole+ll)
382  {
383  for(int t=(int)(m_preData->effectiveMonopole+m+s); t<=(int)(n-1-m_preData->effectiveMonopole+m-ll+s); ++t)
384  {
385  if(t>=0 && t<n)
386  {
387  utilities::DivFactorial<V, P>(n-1-t,n-1-t-m_preData->effectiveMonopole+m-ll+s, &realTmp);
388  *p_list = realTmp;
389  utilities::DivFactorial<V, P>(t, t-m_preData->effectiveMonopole-m-s, &realTmp);
390  (*p_list) *= realTmp;
391  (*p_list).MulBy(m_preData->binomList[(int)((
392  2*m_preData->effectiveMonopole+ll)
393  *m_preData->maxBinom+m_preData->effectiveMonopole+m+s)]
394  *m_preData->binomList[ll*m_preData->maxBinom+s], (*p_list));
395  (*p_list).MulBy(1.0/norm,(*p_list));
396  // This line implements (-1)^t
397  if(t & 1) (*p_list).MulBy(-1.0, (*p_list));
398  // This line implements (-1)^s
399  if(s & 1) (*p_list).MulBy(-1.0, (*p_list));
400  ////////////////////////////////////////////////////////////////////////////////
401  #if _DEBUG_
402  utilities::cout.DebuggingInfo() << "p_list = " << (*p_list).Get() << std::endl;
403  #endif
404  ////////////////////////////////////////////////////////////////////////////////
405  ++p_list;
406  }
407  }

415  ////////////////////////////////////////////////////////////////////////////////
416  //! \brief Performs pre-calculation of elementary symmetric polynomials and writes
417  //! a table of values to symmPolyArray. In the process, also populates zPowStore
418  //! (a list of powers of v/u) and uProd (a list of products u,u*u etc.).
419  ////////////////////////////////////////////////////////////////////////////////
420  void GenerateSymmetricPolys(
421  const int n, //!< Number of particles
422  dcmplx* u, //!< Pointer to the current array of current spinor u coordinates
423  dcmplx* v, //!< Pointer to the current array of current spinor v coordinates
424  utilities::HpWrap<U,P> *zPowStore, //!< Pointer to a table to store powers of z = v/u
425  utilities::HpWrap<U,P> *symmPolyArray, //!< Pointed to a table to store elementary symmetric polynomials
426  utilities::HpWrap<U,P> *uProd) //!< Pointer to a table to store values of u, u*u etc.
427  {
428  // Declare local variables
429  utilities::HpWrap<U,P> tempU[n]; // Temporary wrapper value for the u co-ordinate
430  utilities::HpWrap<U,P> temp; // Temporary value used in internal calculations
431  // To save memory access time, it is convenient to set the ratios of v[j]/u[j] to be the variable z[j].
432  for(int j=0;j<n;j++)
433  {
434  m_z[j].Set(v[j]/u[j]);
435  tempU[j].Set(u[j]);
436  }
437  // Need to multiply all of the Y matrix elements by prod u_i for i!=j and by u_j^n-1-2*m_preData->effectiveMonopole;
438  uProd[0].Set(1.0);
439  for(int j=0; j<n; ++j)
440  {
441  uProd[0] *= tempU[j];
442  }
443  for(int j=1; j<n; ++j)
444  {
445  uProd[j] = uProd[0];
446  }
447  for(int j=0; j<n; ++j)
448  {
449  for(int k=0; k<n-2-2*m_preData->effectiveMonopole; ++k)
450  {
451  uProd[j] *= tempU[j];
452  }
453  // divide by factors of u if n-2-2*m_preData->effectiveMonopole<0
454  for(int k=n-2-2*m_preData->effectiveMonopole; k<0; ++k)
455  {
456  uProd[j] /= tempU[j];
457  }
458  }
459  // Pre-calculate and store powers of v[j]/u[j] for later use
460  for(int j=0; j<n; ++j)
461  {
462  zPowStore[j*n].Set(1.0);
463  }
464  for(int j=0; j<n; ++j)
465  {
466  m_zPow[j] = m_z[j];
467  }
468  for(int j=0; j<n; ++j)
469  {
470  if(j<(int)(n-2*m_preData->effectiveMonopole-1))
471  {
472  for(int i=0; i<n; ++i)
473  {
474  zPowStore[i*n+j+1] = m_zPow[i];
475  }
476  }
477  for(int k=0; k<n; ++k)
478  {
479  m_zPow[k] *= m_z[k];
480  }
481  }
482  // Recursively evaluate elementary symmetric polynomials in n variables
483  // m_esp[k] contains the kth degree elementary symmetric polynomial
484  m_esp[0].Set(1.0);
485  m_esp[1]=m_z[0];
486  for(int k=2; k<n; ++k)
487  {
488  m_esp[k].Set(0.0);
489  }
490  for(int k=2; k<=n; ++k)
491  {
492  for(int j=std::min((int)k, n-1); j>=1; --j)
493  {
494  temp = m_z[k-1];
495  temp *= m_esp[j-1];
496  m_esp[j] += temp;
497  }
498  }
499  // Finally, use e_k(n-1)=e_k(n) - z_j e_k-1(n-1) to convert to an array symmPolyArray[j][i] of esps in n-1
500  // variables. symmPolyArray[j][i] contains the elementary symm. poly. of order i, in n-1 variables
501  // excluding the jth variable.
502  for(int j=0; j<n; ++j)
503  {
504  symmPolyArray[j*n].Set(1.0);
505  }
506  for(int j=0; j<n; ++j)
507  {
508  for(int k=1; k<n; ++k)
509  {
510  temp = m_z[j];
511  temp *= symmPolyArray[j*n+k-1];
512  symmPolyArray[j*n+k] = m_esp[k];
513  symmPolyArray[j*n+k] -= temp;
514  }
515  }
516  ////////////////////////////////////////////////////////////////////////////////
517  #if _DEBUG_
518  utilities::cout.DebuggingInfo() << "\n\tSymm poly array:" << std::endl;
519  for(int j=0; j<n; ++j)
520  {
521  for(int k=1; k<n; ++k)
522  {
523  utilities::cout.DebuggingInfo() << "\t[" << j << "][" << k << "]="
524  << symmPolyArray[j*n+k].Get() << std::endl;
525  }
526  }
527  #endif

532  ////////////////////////////////////////////////////////////////////////////////
533  //! \brief Performs pre-calculation of Jorgen's polynomials and writes
534  //! a table of values to symmPolyArray. In the process, also populates zPowStore
535  //! (a list of powers of v/u).
536  //!
537  ////////////////////////////////////////////////////////////////////////////////
538  void GenerateJorgenPolynomials(
539  const int n, //!< Number of particles
540  dcmplx* u, //!< Pointer to the current array of current spinor u coordinates
541  dcmplx* v, //!< Pointer to the current array of current spinor v coordinates
542  utilities::HpWrap<U,P> *zPowStore, //!< Pointer to a table to store powers of z = v/u
543  utilities::HpWrap<U,P> *symmPolyArray, //!< Pointed to a table to store elementary symmetric polynomials
544  utilities::HpWrap<U,P> *uProd) //!< Pointer to a table to store values of u to some power
545  {
546  utilities::HpWrap<U,P> tempU[n];
547  utilities::HpWrap<U,P> tempV[n];
548  for(int j=0; j<n; ++j)
549  {
550  m_z[j].Set(v[j]/u[j]);
551  tempU[j].Set(u[j]);
552  tempV[j].Set(v[j]);
553  }
554  // Pre-calculate and store powers of v[j]/u[j] for later use
555  for(int j=0; j<n; ++j)
556  {
557  zPowStore[j*n].Set(1.0);
558  }
559  for(int j=0; j<n; ++j)
560  {
561  m_zPow[j] = m_z[j];
562  }
563  for(int j=0; j<n; ++j)
564  {
565  if(j<(int)(n-2*m_preData->effectiveMonopole-1))
566  {
567  for(int i=0; i<n; ++i)
568  {
569  zPowStore[i*n+j+1] = m_zPow[i];
570  }
571  }
572  for(int k=0; k<n; ++k)
573  {
574  m_zPow[k] *= m_z[k];
575  }
576  }
577  // Need to multiply all of the Y matrix elements by u_j^n-1-2*m_preData->effectiveMonopole;
578  for(int j=0; j<n; ++j)
579  {
580  uProd[j].Set(1.0);
581  }
582  for(int j=0; j<n; ++j)
583  {
584  for(int k=0; k<n-1-2*m_preData->effectiveMonopole; ++k)
585  {
586  uProd[j] *= tempU[j];
587  }
588  // divide by factors of u if n-1-2*m_preData->effectiveMonopole<0
589  for(int k=n-1-2*m_preData->effectiveMonopole; k<0; ++k)
590  {
591  uProd[j] /= tempU[j];
592  }
593  }
594  // Recursively evaluate the Jorgen polynomials
595  // Define f^i_t,0 = { 1: t=0
596  // { 0: t>=1
597  for(int i=0; i<n; ++i)
598  {
599  symmPolyArray[i*n].Set(1.0);
600  for(int t=1; t<n; ++t)
601  {
602  symmPolyArray[i*n+t].Set(0.0);
603  }
604  }
605  // Now go through from j=1 to N and set f^i_t,j = { f^i_{t,j-1} j=i
606  // { f^i_{t,j-1}u_j j!=i and t=0
607  // { f^i_{t,j-1}u_j + f^i_{t-1,j-1} v_n j!=i and t>0
608  utilities::HpWrap<U,P> temp;
609  utilities::HpWrap<U,P> temp2;
610  for(int j=0; j<n; ++j)
611  {
612  for(int i=0; i<n; ++i)
613  {
614  if(i!=j)
615  {
616  // i!=j and t>0 case
617  for(int t=n-1; t>0; --t)
618  {
619  temp = symmPolyArray[i*n+t];
620  temp *= tempU[j];
621  temp2 = symmPolyArray[i*n+t-1];
622  temp2 *= tempV[j];
623  temp2 += temp;
624  symmPolyArray[i*n+t] = temp2;
625  }
626  // i!=j and t=0 case
627  symmPolyArray[i*n] *= tempU[j];
628  }
629  }
630  }
631  ////////////////////////////////////////////////////////////////////////////////
632  #if _DEBUG_
633  utilities::cout.DebuggingInfo() << "\n\tJorgen poly array:" << std::endl;
634  for(int j=0; j<n; ++j)
635  {
636  for(int k=0; k<n; ++k)
637  {
638  utilities::cout.DebuggingInfo() << "\t[" << j << "][" << k << "]="
639  << symmPolyArray[j*n+k].Get() << std::endl;
640  }
641  }
642  #endif
643  ////////////////////////////////////////////////////////////////////////////////
644  }

646  ////////////////////////////////////////////////////////////////////////////////
647  //! \brief Pre calculation of the "p-functions" pArray and also z=v/u.
648  //!
649  //! See the relevant appendix of J.K. Jain's book on composite fermions
650  //! for the definition of these functions.
651  ////////////////////////////////////////////////////////////////////////////////
652  void GeneratePfunctions(
653  const int n, //!< Number of particles
654  const int nbrLLs, //!< Number of Landau levels
655  dcmplx* u, //!< Pointer to the current array of current spinor u coordinates
656  dcmplx* v, //!< Pointer to the current array of current spinor v coordinates
657  utilities::HpWrap<U,P> *pArray, //!< Pointer to a table of "p-functions" to be populated
658  utilities::HpWrap<U,P> *z) //!< Pointer to an array containing v/u to be populated
659  {
660  utilities::HpWrap<U,P> tmp1,tmp2,tmp3;
661  utilities::HpWrap<U,P> runTot,runTot2;
662  utilities::HpWrap<U,P> *p_fArray;
663  utilities::HpWrap<U,P> *p_pArray;
664  for(int j=0; j<n; ++j)
665  {
666  z[j].Set(v[j]/u[j]);
667  }
668  // First, determine f_j(alpha, beta) functions
669  for(int a=0; a<nbrLLs; ++a)
670  {
671  for(int b=0; b<nbrLLs; ++b)
672  {
673  if((a==0 && b==0) || (a+b>=nbrLLs))
674  {}
675  else
676  {
677  for(int j=0; j<n; ++j)
678  {
679  runTot.Set(0.0);
680  // Store z_j
681  tmp1=*(z+j);
682  for(int k=0;k<n;k++)
683  {
684  if(k!=j)
685  {
686  // Store z_k
687  tmp2 = *(z+k);
688  // Store z_k - z_j
689  tmp3 = tmp2;
690  tmp3 -= tmp1;
691  runTot2.Set(1.0);
692  for(int i=0; i<(a+b); ++i)
693  {
694  if(i<a)
695  {
696  // calculate z_k^a
697  runTot2 *= tmp2;
698  }
699  // Calculate 1/(z_k-z_j)^(a+b)
700  runTot2 /= tmp3;
701  }
702  // Sum over k
703  runTot += runTot2;
704  }
705  }
706  // Implement (-1)^b
707  if(b&1)
708  {
709  runTot.MulBy(-1,runTot);
710  }
711  tmp1.Set(*(u+j));
712  // Divide by u_j^(a+b)
713  for(int i=0; i<(a+b); ++i)
714  {
715  runTot /= tmp1;
716  }
717  // Set pointer to the correct row of m_fArray
718  p_fArray = m_fArray+(a*nbrLLs+b)*n;
719  // Store array of values
720  *(p_fArray+j) = runTot;
721  }
722  }
723  }
724  }
725  ////////////////////////////////////////////////////////////////////////////////
726  #if _DEBUG_
727  //output matrix elements
728  for(int a=0; a<nbrLLs; ++a){
729  for(int b=0; b<nbrLLs; ++b){
730  if((a==0 && b==0) ||(a+b>=nbrLLs))
731  {}
732  else
733  {
734  for(int j=0; j<n; ++j){

1005  //!

1007  //! for a given set of co-ordinates.
1008  //!
1009  //! There are two steps: first we populate a n x n Slater matrix with
1010  //! lowest Landau level projected monopole harmonics, then we calculate
1011  //! the log of its Slater determinant.
1012  //!

1014  ////////////////////////////////////////////////////////////////////////////////
1015  dcmplx PolarisedNEF(
1016  const int n, //!< Number of particles
1017  const int nbrLLs, //!< Number of Landau levels in the construction
1018  dcmplx* u, //!< Pointer to the current array of current spinor u coordinates
1019  dcmplx* v) //!< Pointer to the current array of current spinor v coordinates

1026 
1027  // Generate elementary symmetric polynomials and prod u_i for i!=j * u_j^n-1-2*m_preData->effectiveMonopole
1028  //this->GenerateSymmetricPolys(n,u,v,m_zPowStore,m_symmPolyArray,m_uProd);

1031  utilities::HpWrap<U,P> term;

1699  for(int i=0; i<n; ++i)
