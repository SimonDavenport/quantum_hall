31 ////////////////////////////////////////////////////////////////////////////////
32 //! \brief Generate the wave function label for a given file
33 //!
34 //! \return Wave function label
35 ////////////////////////////////////////////////////////////////////////////////
36 inline std::string GenerateWaveFunctionName(
37  const FQHE::CompositeFermionData* cfData, //!< Pointer to composite fermion wave function data
38  const FQHE::WaveFunctionData* wfData) //!< Pointer to wave function data
39 {
40  if(wfData->type == FQHE::_LAUGHLIN_)
41  {
42  return "laughlin_";
43  }
44  else if(wfData->type == FQHE::_COMPOSITE_FERMION_)
45  {
46  if(cfData->NEF)
47  {
48  return "nef_";
49  }
50  else
51  {
52  return "pef_";
53  }
54  }
55  else if(wfData->type == FQHE::_BONDERSON_SLINGERLAND_)
56  {
57  return "bonderson_slingerland_";
58  }
59  else if(wfData->type == FQHE::_MOORE_READ_)
60  {
61  return "moore_read_";
62  }
63  else
64  {
65  return "";
66  }
67 }

70 //! \brief Generate the base file name labelling the output data
71 //!
72 //! \return base file name
73 ////////////////////////////////////////////////////////////////////////////////
74 inline std::string GenerateBaseFileName(
75  const GeneralOptions* parameters, //!< Pointer to set of program parameters
76  const FQHE::CompositeFermionData* cfData, //!< Pointer to composite fermion wave function data
77  const FQHE::WaveFunctionData* wfData) //!< Pointer to wave function data
78 {
79  std::stringstream name;
80  name.str("");
81  name << parameters->path << "/" << GenerateWaveFunctionName(cfData, wfData);
82  name << wfData->fillNumerator << "_" << wfData->fillDenominator << "_n_" << wfData->nbr
83  << "_size_" << parameters->rows << "x" << parameters->columns;
84  return name.str();
85 }
