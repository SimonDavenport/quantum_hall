7 //! \file
8 //! This program is designed to generate (approximately) the real space
9 //! entanglement spectrum of a given fractional quantum Hall wave function.
10 //!
11 //! The calculation performed by this code is roughly as follows.
12 //!
13 //! We want to determine an approximation to the reduced density
14 //! matrix for a quantum Hall system subdivided into parts A and B
15 //! in real space or in particle space. The entanglement spectrum of
16 //! the associated reduced matrix is a plot of \f$ - \f$ log(eigenvalues) vs.
17 //! the angular momentum sector.
18 //!
19 //! To do that, we can obtain an approximation to the spectrum from a
20 //! certain "pseudo" reduced density matrix. First define:
21 //!
22 //! \f[ \Phi(L_z^A,Z_A,Z_B) = \int d \phi exp(-2I\pi L_z^A \phi) \psi(z_1 exp(2I\pi \phi),\ldots,z_k exp(2I\pi \phi), z_k+1,\ldots z_N) \f]
23 //!
24 //! The Fourier transform in \f$ \Phi \f$ selects a specific angular momentum
25 //! sector in the wave function. The reduced density matrix is given by:
26 //!
27 //! \f[ \rho_A(Z_A;Z_A')=\int dZ_B \Phi(L_z^A,Z_A,Z_B) \Phi^*(L_Z^A,Z'_A,Z_B) \f]
28 //!
29 //! where \f$ Z_A \f$ and \f$ Z_B \f$ are the sets of coordinates in the A and B subsystems
30 //!
31 //! For the particle entanglement spectrum \f$ Z_A \f$ and \f$ Z_B \f$ can take any value
32 //! but for a real space entanglement spectrum (RSES) the \f$ Z_A \f$ and \f$ Z_B \f$
33 //! co-ordinates must lie in separate regions.
34 //!
35 //! An approximation to rho_A is given by generating a set of rho_A
36 //! using a selected list of sample coordinate values. We shall refer to
37 //! this list as the "indexes". The same indexes are used for \f$ Z_A \f$ and \f$ Z_A' \f$,
38 //! but there is no restriction for the number of rows and columns in
39 //! the \f$ \rho_A \f$ to be equal. The number of indexes must be greater than both
40 //! the rows and columns of the reduced density matrix. As the size of
41 //! the effective reduced density matrix is increased, so its eigenvalues
42 //! tend to converge to those of the true reduced density matrix. The
43 //! convergences expected to be faster for a rectangular effective \f$ \rho_A \f$
44 //! (for which we then need to calculate the SVD).
45 //!
46 //! When using this method, it is very important to select a good set of
47 //! indexes. The co-ordinate samples used must have a good overlap with
48 //! the angular momentum sectors for which we want to sample.
49 //! The indexes are selected by taking \f$ l^i_A \f$ to be an integer closest to
50 //! \f$ |z_i|^2 \f$ and then only selecting samples that satisfy:
51 //!
52 //! \f$ \sum_i l^i_A = L_z^A \f$
53 //!
54 //! Further, when two such samples share the same set of \f$ l^i_A \f$ we can
55 //! discard one of them without loss of rank. Also, we don't need to
56 //! include all possible \f$ l_z^A \f$ sets to get good convergence.
57 //!
58 //! Of particular interest is the low lying part of the RSES (the largest)
59 //! eigenvalues of \f$ \rho_A \f$. In that case, we can apply a further constraint
60 //! that the particles in subsystem A are located in a disk of discRadius \f$ r_A \f$
61 //! (since for the RSES calculation, the particles are confined to some
62 //! region of space).

1072 ////////////////////////////////////////////////////////////////////////////////
1073 //! \brief This function determines the value of lzA associated with the
1074 //! lowest lzA sector that appears in the entanglement spectrum (RSES).
1075 //!
1076 //! This is deduced by considering the orbital angular momentum
1077 //! of the CF-EWF quasihole wave function with the lowest orbitals occupied
1078 //!
1079 //! - Laughlin case
1080 //!
1081 //! Example quasihole diagram of lowest lzA state
1082 //! \verbatim
1083 //! X X X X X _ _ _ _ _ * X X X X X _ _ _ _ _ * ...
1084 //! \endverbatim
1085 //!
1086 //! - Composite fermion case
1087 //!
1088 //! Example quasihole diagram of lowest lzA state
1089 //! \verbatim
1090 //! X X X _ _ _ * X X X X X _ _ _ _ _ * X X X X X _ _ _ _ _
1091 //! X X _ _
1092 //! \endverbatim
1093 //!
1094 //! \return The lzA value of the lowest sector in the RSES
1095 //!
1096 //! \todo Update to include Moore-Read and BS cases
1097 //!
1098 ////////////////////////////////////////////////////////////////////////////////
1099 
1100 double FindStartSector(
1101  const GeneralOptions* parameters, //!< Pointer to parameter set
1102  const FQHE::WaveFunctionData* wfData, //!< A given wave function data set
1103  const FQHE::CompositeFermionData* cfData) //!< A given composite fermion data set
1104 {
1105  int branchLza=0; // This variable contains an integer which is twice
1106  // the orbital angular momentum
1107 
1108  if(wfData->type==FQHE::_LAUGHLIN_) // Laughlin case
1109  {
1110  for(int i=0;i<parameters->nbrFixed;i++)
1111  {
1112  branchLza+=-(wfData->nbr-1)+2*i;
1113  }
1114 
1115  branchLza*=wfData->jastrowExponent;
1116  }
1117 
1118  if(wfData->type==FQHE::_COMPOSITE_FERMION_) // Composite fermion case
1119  {
1120  // Flux attached part
1121 
1122  for(int i=0;i<parameters->nbrFixed;i++)
1123  {
1124  branchLza+=-(wfData->nbr-1)+2*i;
1125  }
1126 
1127  branchLza*=2*wfData->jastrowExponent;
1128 
1129  // Composite fermion LLs part
1130 
1131  int lzList[wfData->nbr]; // list of lzA values of each CF nbrLLs orbital
1132  int *p_list=lzList; // pointer to a value in the list
1133 
1134  double Q=(double)((int)wfData->nbr-cfData->LLup*cfData->LLup-cfData->LLdown*cfData->LLdown)/(2*cfData->LLup+2*cfData->LLdown);
1135 
1136  for(int l=0;l<cfData->LLup;l++)
1137  {
1138  int degen=(int)(2.0*(Q+l))+1;
1139 
1140  for(int i=0;i<degen;i++)
1141  {
1142  *(p_list)=-(degen-1)+2*i;
1143  p_list++;
1144  }
1145  }
1146 
1147  // Sort the list in terms of angular momentum values
1148 
1149  for(int i=0;i<wfData->nbr;i++)
1150  {
1151  for(int j=i;j<wfData->nbr;j++)
1152  {
1153  if(lzList[i]>lzList[j])
1154  {
1155  int swap=lzList[i];
1156  lzList[i]=lzList[j];
1157  lzList[j]=swap;
1158  }
1159  }
1160  }
1161 
1162  // Occupy the lowest lzA set of orbitals
1163 
1164  if(cfData->NEF)
1165  {
1166  for(int i=0;i<parameters->nbrFixed;i++)
1167  {
1168  branchLza-=lzList[i];
1169  }
1170  }
1171  else
1172  {
1173  for(int i=0;i<parameters->nbrFixed;i++)
1174  {
1175  branchLza+=lzList[i];
1176  }
1177  }
1178  }
1179 
1180  return (double)branchLza/2.0;
1181 }
