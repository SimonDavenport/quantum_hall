121  int variableSize = bufferSize;

123  MPI_Get_processor_name(host, &variableSize);

126  {
127  recvBuffer = new (std::nothrow) char[m_nbrProcs*bufferSize];
128  }
129  MPI_Gather(host, bufferSize, MPI_CHAR, recvBuffer, bufferSize, MPI_CHAR, 0, m_comm);
130  if(0 == m_id) // For the master node
131  {
132  for(int i=0; i<m_nbrProcs; ++i)
133  {
134  memcpy(host, recvBuffer+bufferSize*i, bufferSize*sizeof(char));
135  m_hosts.push_back(host);
136  }
137  }
138  if(0 != recvBuffer)
139  {
140  delete[] recvBuffer;
141  }
142  return;
143  }

232  //! a roughly equal memory distribution to store a triagular matrix in parallel)
233  //!

241  //!

243  //! Processor Tasks Task Task
244  //!
245  //! 0 60 0 59
246  //! 1 71 60 130
247  //! 2 93 131 223
248  //! 3 226 224 449
249  //!
250  //! \endverbatim

278  }

282  int hi,lo;

285  {

293  taskRemain-=taskProc;
294  weight = 0.5*taskRemain*(taskRemain+1.0);
295  lo=hi+1;
