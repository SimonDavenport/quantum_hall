34  FQHE::WaveFunctionData *wavefunctionData) //!< Address of a struct of wave function data
35  :

37  m_nCluster(0),
38  m_nLastCluster(0),
39  m_pfaff(0)
40 {
41  m_wfData->fillNumerator = m_wfData->nbrClusters;
42  m_wfData->fillDenominator = m_wfData->nbrClusters*m_wfData->jastrowExponent+2;
43  m_wfData->shift = m_wfData->jastrowExponent+2;
44  if(m_wfData->nbr%m_wfData->nbrClusters!=0)
45  {
46  std::cerr << "\n\tERROR in FQHE::Parafermion. Invalid particle number for cluster size" << std::endl;
47  exit(EXIT_FAILURE);
48  }
49 
50  if(m_wfData->nbr>64)
51  {
52  std::cerr << "\n\tERROR in FQHE::Parafermion. Cannot use more than 64 particles due to 64-bit binary implementation" << std::endl;
53  exit(EXIT_FAILURE);
54  }
55  {
56  std::stringstream temp;
57  temp.str("");
58  temp << "Z_" << m_wfData->nbrClusters << "_Parafermion";
59  m_wfData->wfName = temp.str();
60  }
61  m_wfData->statistics = _FERMIONS_;
62  m_wfData->fillingFactor =(double)m_wfData->fillNumerator/m_wfData->fillDenominator;
63  if(m_wfData->geometry==_SPHERE_ || m_wfData->geometry==_DISC_)
64  {
65  m_wfData->flux = (double)m_wfData->nbr/(m_wfData->fillingFactor)-m_wfData->shift;
66  m_wfData->monopoleStrength = (m_wfData->nbr*m_wfData->fillDenominator/m_wfData->fillNumerator-m_wfData->shift)/2.0;
67  }
68  m_wfData->GenerateFileName();
69  m_wfData->InitRadius();
70  m_wfData->CheckAndPrint();
71  // Generate all independent permutations of index labels. Only labels swapped between
72  // different clusters will contribute
73  m_nCluster = m_wfData->nbr/m_wfData->nbrClusters;
74  // Note that we can exploit a factorization in terms of Pfaffian factors to reduce
75  // the number of clusters by a factor of 2 (for even number of clusters) or
76  // by a factor of 2 excluding the final cluster (for odd number of clusters)
77  // For odd number of clusters, the last cluster will remain the same size
78  m_nLastCluster = m_wfData->nbrClusters&1 ? m_nCluster : 0;
79  // Increase the number of particles in each remaining cluster by a factor of 2
80  m_nCluster *= 2;
81  // Decrease the number of clusters by the appropriate factor (not including the last one)
82  m_wfData->nbrClusters = (m_wfData->nbrClusters)/2;
83  // Allocate memory to construct a Pfaffian for each new cluster (not including the last one)
84  m_pfaff = new dcmplx[m_nCluster*m_nCluster];
85  // Recursively generate subsets of the cluster permutations
86  std::vector<unsigned char> originalList(m_wfData->nbr);
87  for(unsigned char i=0; i<m_wfData->nbr; ++i)
88  {
89  originalList[i] = i;
90  }
91  std::vector<unsigned int> nClusterList;
92  for(int i=0; i<m_wfData->nbrClusters; ++i)
93  {
94  nClusterList.push_back(m_nCluster);
95  }
96  if(m_nLastCluster)
97  {
98  nClusterList.push_back(m_nLastCluster);
99  }
100  // Allocate temporary working memory
101  std::vector<unsigned char> currPerm(m_wfData->nbr);
102  this->BuildSubset(m_wfData->nbr, &nClusterList, &originalList, &currPerm, &m_clusterPermutations, 0);
103  std::cout << "\tTotal number of cluster permutations = " << m_clusterPermutations.size()
104  << std::endl << std::endl;

107 //!
108 //! Recursively generate unique subsets of the particle indices in
109 //! different clusters
110 //!
111 void FQHE::Parafermion::BuildSubset(
112  const unsigned int nbrRemain, //!< Number of indices remaining to be assigned to clusters
113  std::vector<unsigned int>* nClusterList,
114  //!< List of the number of particles in each cluster
115  std::vector<unsigned char>* remainList,
116  //!< List of indexes to be assigned to clusters
117  std::vector<unsigned char>* currPerm,
118  //!< Current permutation being constructed (of size nbr)
119  std::vector<std::vector<unsigned char> >* clusterPermutations,
120  //!< Full list of cluster permutations
121  const unsigned int level) //!< Recursion level (= which
122  //! cluster we're assigning)
123  const
124 {
125  unsigned int nCluster = (*nClusterList)[level];
126  uint64_t mask = utilities::binary::FirstBinaryHammingNumber64(nCluster);
127  const unsigned int nbrSubsets = utilities::BinomialFromTable(nbrRemain,nCluster);
128  unsigned int offset = 0;
129  for(unsigned int i=0; i<level; ++i)
130  {
131  offset += (*nClusterList)[i];
132  }
133  // Generate subsets of the original list by masking it with a list of binary Hamming
134  // numbers in lexicographic order
135  for(unsigned int i=0; i<nbrSubsets; ++i)
136  {
137  uint64_t temp = mask;
138  auto it = currPerm->begin()+offset;
139  while(temp!=0)
140  {
141  // Isolate the right-most bit using a 2s complement, then
142  // use its position to mask the original list
143  uint64_t occupied = temp & -temp;
144  *it = (*remainList)[utilities::binary::HammingWeight64(occupied-1)];
145  // Remove right-most bit and proceed to the next iteration
146  temp = temp ^ occupied;
147  ++it;
148  }
149  // Put remaining unused indices in a separate list
150  temp = (~mask) & utilities::binary::FirstBinaryHammingNumber64(nbrRemain);
151  std::vector<unsigned char> subRemainList(nbrRemain-nCluster);
152  it = subRemainList.begin();
153  while(temp!=0)
154  {
155  uint64_t occupied = temp & -temp;
156  *it = (*remainList)[utilities::binary::HammingWeight64(occupied-1)];
157  temp = temp ^ occupied;
158  ++it;
159  }
160  if(level == nClusterList->size()-1)
161  {
162  clusterPermutations->push_back(*currPerm);
163  }
164  else
165  {
166  this->BuildSubset(nbrRemain-nCluster,nClusterList,&subRemainList,currPerm,clusterPermutations,level+1);
167  }
168  mask = utilities::binary::NextHammingNumber64(mask);
169  }
170 }

197  // the number of clusters by a factor of roughly 2.

199  {

202  for(int a=0;a<m_wfData->nbrClusters;++a)
