77 /**
78  * Initializes the Mersenne Twister with a seed.
79  *
80  * \param s seed
81  */
82 void MersenneTwister::init_genrand(unsigned long s)
83 {
84  mt_[0]= s & 0xffffffffUL;
85  for (mti_=1; mti_<N; mti_++) {
86  mt_[mti_] =
87  (1812433253UL * (mt_[mti_-1] ^ (mt_[mti_-1] >> 30)) + mti_);
88  /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
89  /* In the previous versions, MSBs of the seed affect */
90  /* only MSBs of the array mt_[]. */
91  /* 2002/01/09 modified by Makoto Matsumoto */
92  mt_[mti_] &= 0xffffffffUL;
93  /* for >32 bit machines */
94  }
95  // Store the seed
96  s_ = s;
97  seeded_by_array_ = false;
98  seeded_by_int_ = true;
99 }

101 /**
102  * Seed the Mersenne Twister using an array.
103  *
104  * \param init_key an array for initializing keys
105  * \param key_length the length of \a init_key
106  */
107 void MersenneTwister::init_by_array(unsigned long* init_key, int key_length)
108 {
109  // Store the key array
110  int i, j, k;
111  init_genrand(19650218UL);
112  i=1; j=0;
113  k = (N>key_length ? N : key_length);
114  for (; k; k--) {
115  mt_[i] = (mt_[i] ^ ((mt_[i-1] ^ (mt_[i-1] >> 30)) * 1664525UL))
116  + init_key[j] + j; /* non linear */
117  mt_[i] &= 0xffffffffUL; /* for WORDSIZE > 32 machines */
118  i++; j++;
119  if (i>=N) { mt_[0] = mt_[N-1]; i=1; }
120  if (j>=key_length) j=0;
121  }
122  for (k=N-1; k; k--) {
123  mt_[i] = (mt_[i] ^ ((mt_[i-1] ^ (mt_[i-1] >> 30)) * 1566083941UL))
124  - i; /* non linear */
125  mt_[i] &= 0xffffffffUL; /* for WORDSIZE > 32 machines */
126  i++;
127  if (i>=N) { mt_[0] = mt_[N-1]; i=1; }
128  }
129 
130  mt_[0] = 0x80000000UL; /* MSB is 1; assuring non-zero initial array */
131 
132  // Store the seed
133  if (init_key_ != NULL) {
134  delete[] init_key_;
135  }
136  init_key_ = new unsigned long[key_length];
137  for (int k = 0; k < key_length; k++) {
138  init_key_[k] = init_key[k];
139  }
140  key_length_ = key_length;
141  seeded_by_int_ = false;
142  seeded_by_array_ = true;
143 }

145 /**
146  * Generates a random number on [0,0xffffffff]-interval
147  *
148  * \return random number on [0, 0xffffffff]
149  */
150 unsigned long MersenneTwister::genrand_int32(void)
151 {
152  unsigned long y;
153  static unsigned long mag01[2]={0x0UL, MATRIX_A};
154  /* mag01[x] = x * MATRIX_A for x=0,1 */
155 
156  if (mti_ >= N) { /* generate N words at one time */
157  int kk;
158 
159  if (mti_ == N+1) /* if init_genrand() has not been called, */
160  init_genrand(5489UL); /* a default initial seed is used */
161 
162  for (kk=0;kk<N-M;kk++) {
163  y = (mt_[kk]&UPPER_MASK)|(mt_[kk+1]&LOWER_MASK);
164  mt_[kk] = mt_[kk+M] ^ (y >> 1) ^ mag01[y & 0x1UL];
165  }
166  for (;kk<N-1;kk++) {
167  y = (mt_[kk]&UPPER_MASK)|(mt_[kk+1]&LOWER_MASK);
168  mt_[kk] = mt_[kk+(M-N)] ^ (y >> 1) ^ mag01[y & 0x1UL];
169  }
170  y = (mt_[N-1]&UPPER_MASK)|(mt_[0]&LOWER_MASK);
171  mt_[N-1] = mt_[M-1] ^ (y >> 1) ^ mag01[y & 0x1UL];
172 
173  mti_ = 0;
174  }
175 
176  y = mt_[mti_++];
177 
178  /* Tempering */
179  y ^= (y >> 11);
180  y ^= (y << 7) & 0x9d2c5680UL;
181  y ^= (y << 15) & 0xefc60000UL;
182  y ^= (y >> 18);
183 
184  return y;
185 }
