35  //! \brief Hamming weight algorithm with minimal number of arithmetic operations
36  //!
37  //! THIS ALGORITHM IS ADAPTED FROM THE WIKIPEDIA ARTICLE:
38  //! http://en.wikipedia.org/wiki/Hamming_weight
39  //!
40  //! x86 implementation requires 24 instructions
41  //!
42  //! Example:
43  //!
44  //! x = 00 11 01 11 11 01 11 00 10 11 11 11 11 01 01 01
45  //!
46  //! (x>>1) & 010101010101... = 00 01 00 01 01 00 01 00 01 01 01 01 01 00 00 00
47  //!
48  //! This function maps bit pairs in the following way:
49  //!
50  //! 11 -> 01 10 -> 01 01 -> 00 00 -> 00
51  //!
52  //! Then x -= (x>>1) & 010101010101... gives
53  //!
54  //! 00 10 01 10 10 01 10 00 01 10 10 10 10 01 01 01
55  //! i.e. 0 2 1 2 2 1 2 0 1 2 2 2 2 1 1 1
56  //!
57  //! This gives the count of sets bits from each original bit pair
58  //!
59  //! (x & m2) + ((x >> 2) & m2) simply adds the numbers stored in pairs
60  //! of 2-bits using that m2 = 001100110011...)
61  //!
62  //! NOTE (x + (x >> 2)) & m2 won't work because there can be 4 set bits
63  //! in each 4-bit section - this is represetned by 0100, but this would
64  //! by masked off by m2
65  //!
66  //! 00 10 00 10 00 01 00 00 00 10 00 10 00 01 00 01
67  //! + 00 00 00 01 00 10 00 10 00 01 00 10 00 10 00 01
68  //! 0010 0011 0011 0010 0011 0100 0011 0010
69  //! i.e. 2 3 3 2 3 4 3 2
70  //!
71  //! The remaining steps are in the same spirit
72  //!
73  //! (x + (x >> 4)) & 0000111100001111...
74  //!
75  //! 0010 0011 0011 0010 0011 0100 0011 0010
76  //! + xxxx 0010 0011 0011 0010 0011 0100 0011
77  //! xxxx 0101 0110 0101 0101 0111 0111 0101
78  //!
79  //! Then take & 0000111100001111...
80  //!
81  //! 00000101 00000101 00000111 00000101
82  //! i.e. 2+3=5 2+3=5 3+4=7 2+3=5
83  //!
84  //! Next we use a trick of multiplying by h01 = 0x0101010101010101
85  //!
86  //! x * h01 = 0001011000010110000101100001011000010110000100010000110000000101
87  //!
88  //! Finally, right shift by 56 (i.e. count only the left 8 bits)
89  //!
90  //! return 00010110 = 22=5+5+7+5
91  //!
92  //! \return Hamming weight (number of 1 bits in the number)
93  //!
94  ////////////////////////////////////////////////////////////////////////////////
95  int HammingWeight64(
96  uint64_t x) //!< Number to find the Hamming weight of
97  {
98  if(0==x) return 0;
99  x -= (x >> 1) & m1; // Put count of each 2 bits into those 2 bits
100  x = (x & m2) + ((x >> 2) & m2); // Put count of each 4 bits into those 4 bits
101  x = (x + (x >> 4)) & m4; // Put count of each 8 bits into those 8 bits
102  return (x * h01)>>56; //returns left 8 bits of x + (x<<8) + (x<<16) + (x<<24) + ...
103  }

105  ////////////////////////////////////////////////////////////////////////////////
106  //! \brief Iterative Hamming weight algorithm.
107  //!
108  //! This algorithm is more efficient when the number of bits counted
109  //! is known to be very small.
110  //!
111  //! x86 implementation requires 13 instructions
112  //!
113  //! Example:
114  //!
115  //! x = 00 00 01 10 00 00 00 10 00 00 01 00 00 00 01 00
116  //!
117  //! x-1 = 00 00 01 10 00 00 00 10 00 00 01 00 00 00 00 11
118  //!
119  //! x &= x-1 gives 00 00 01 10 00 00 00 10 00 00 01 00 00 00 00 00
120  //!
121  //! x-1 = 00 00 01 10 00 00 00 10 00 00 01 11 11 11 11 11
122  //!
123  //! x &= x-1 gives 00 00 01 10 00 00 00 10 00 00 00 00 00 00 00 00
124  //!
125  //! x-1 = 00 00 01 10 00 00 00 01 11 11 11 11 11 11 11 11
126  //!
127  //! x &= x-1 gives 00 00 01 10 00 00 00 00 00 00 00 00 00 00 00 00
128  //!
129  //! etc.
130  //!
131  //! Notice that the x &= x-1 masks off all bits to the right of the right
132  //! -most set bit. This operation needs to be completed M times, where M
133  //! is the number of set bits
134  //!
135  //! \return Hamming weight (number of 1 bits in the number)
136  ////////////////////////////////////////////////////////////////////////////////
137  int HammingWeight64Iterative(
138  uint64_t x) //!< Number to find the Hamming weight of
139  {
140  if(0==x) return 0;
141  unsigned int i=1;
142  while (x &= (x-binary::number1)) ++i;
143  return i;
144  }

146  ////////////////////////////////////////////////////////////////////////////////
147  //! \brief Round down to the nearest power of 2 or in other words isolate the
148  //! left-most bit
149  //!
150  //! x86 implementation requires 22 instructions
151  //!
152  //! Example:
153  //!
154  //! x = 00 00 01 10 00 00 00 10 00 00 01 00 00 00 01 00
155  //!
156  //! x |= x >> 1 gives 00 00 01 11 00 00 00 11 00 00 01 10 00 00 01 10
157  //!
158  //! x |= x >> 2 gives 00 00 01 11 11 00 00 11 11 00 01 11 10 00 01 11
159  //!
160  //! x |= x >> 4 gives 00 00 01 11 11 11 11 11 11 11 11 11 11 11 11 11
161  //!
162  //! The remaining operations do not change this arrangement. Then the
163  //! last steps are to add on and right shift by 1, thereby isolating the
164  //! left-most bit
165  //!
166  //! \return Input rounded down to the nearest power of 2
167  ////////////////////////////////////////////////////////////////////////////////
168  uint64_t IsolateLeftMostSetBit(
169  uint64_t x) //!< A number to find the left-most set bit of
170  {
171  x |= x >> 1;
172  x |= x >> 2;
173  x |= x >> 4;
174  x |= x >> 8;
175  x |= x >> 16;
176  x |= x >> 32;
177  ++x;
178  return x >> binary::number1;
179  }

212  ////////////////////////////////////////////////////////////////////////////////
213  //! \brief Generate the next lexicographic permutation of numbers with the
214  //! same Hamming weight
215  //!
216  //! Algorithm derived from:
217  //!
218  //! http://www.geeksforgeeks.org/next-higher-number-with-same-number-of-set-bits/
219  //!
220  //! EXAMPLE:
221  //!
222  //! 10011100
223  //! 00011100 - right most string of 1's in x
224  //! 00000011 - right shifted pattern except left most bit ------> [A]
225  //! 00010000 - isolated left most bit of right most 1's pattern
226  //! 00100000 - shiftleft-ed the isolated bit by one position ------> [B]
227  //! 10000000 - left part of x, excluding right most 1's pattern ------> [C]
228  //! 10100000 - add B and C (OR operation) ------> [D]
229  //! 10100011 - add A and D to get the final result
230  //!
231  //! x86 implementation requires 15 instructions
232  //!
233  //! Example:
234  //!
235  //! x = 00 00 00 00 00 00 00 00 00 00 00 00 00 10 01 11 00
236  //!
237  //! -x gives the 2s complement (flip bits, add 1)
238  //!
239  //! temp = x & -x = 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 01 00
240  //!
241  //! x + temp = 00 00 00 00 00 00 00 00 00 00 00 00 00 10 10 00 00 [D]
242  //!
243  //! x ^ (x + temp) = 00 00 00 00 00 00 00 00 00 00 00 00 00 00 11 11 00
244  //!
245  //! x ^ (x + temp) /temp = 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 11 11
246  //!
247  //! previous line >> 2 = 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 11
248  //!
249  //! OR x ^ (x + temp) >> 2+log_2 temp gives the same result
250  //!
251  //! final output is 00 00 00 00 00 00 00 00 00 00 00 00 00 10 10 00 11
252  //!
253  //! \return The lexicographically next binary number of the same Hamming weight
254  ////////////////////////////////////////////////////////////////////////////////
255  uint64_t NextHammingNumber64(
256  const uint64_t x) //!< A number with some bitwise representation
257  {
258  if(0==x) return x;
259  uint64_t temp = (x & -x); // Isolates the right-most bit
260  return ((x ^ (x + temp)) / temp) >> 2 | (x + temp);
261  }

263  ////////////////////////////////////////////////////////////////////////////////
264  //! \brief Generate the index in the list of lexicographically ordered Hamming
265  //! numbers assuming all possible numbers are included in order
266  //!
267  //! \return The position of the input in a lexicographically ordered list of
268  //! hamming numbers
269  ////////////////////////////////////////////////////////////////////////////////
270  uint64_t IndexHammingNumber64(
271  const unsigned int hammingWeight, //!< Hamming weight of the set
272  const uint64_t x) //!< A number that we want to find the index of
273  {
274  uint64_t output = 0;
275  uint64_t temp = x;
276  for(int p = hammingWeight; p>0; --p)
277  {
278  uint64_t leftMostBit = IsolateLeftMostSetBit(temp);
279  int m = Base2Log(leftMostBit);
280  if(p<=m)
281  {
282  output += utilities::BinomialFromTable(m, p);
283  }
284  temp ^= leftMostBit;
285  }
286  return output;
287  }

289  ////////////////////////////////////////////////////////////////////////////////
290  //! \brief Generate the Hamming number with the given hamming weight and at
291  //! a given position in a lexicographic table (without having to generate
292  //! the full table of lexigographic permutations). This assumes a full list
293  //! of numbers is going to be indexed
294  //!
295  //! \return The Hamming number at the index returnIndex
296  ////////////////////////////////////////////////////////////////////////////////
297  uint64_t GenerateHammingNumber(
298  const unsigned int hammingWeight, //!< Hamming weight of the set
299  const uint64_t index) //!< Index to return the Hamming number for
300  {
301  if(0 == index)
302  {
303  return FirstBinaryHammingNumber64(hammingWeight);
304  }
305  else
306  {
307  // Determine the exact binary Hamming number corresponding to this
308  // index (assuming lexicographic ordering)
309  uint64_t returnVal = 0;
310  uint64_t currIndex = 0;
311  for(int p = hammingWeight-1; p>=0; --p)
312  {
313  // Determine the position of the left-most bit
314  // then next left most bit etc.
315  int m = p;
316  uint64_t incrementIndex;
317  do
318  {
319  incrementIndex = utilities::BinomialFromTable(m, p);
320  currIndex += incrementIndex;
321  ++m;
322  }
323  while(index >= currIndex);
324  // We've gone one step too far here, so go back one step
325  currIndex -= incrementIndex;
326  // Set the next left-most bit in the return value
327  returnVal |= (binary::number1 << (m-1));
328  }
329  return returnVal;
330  }
