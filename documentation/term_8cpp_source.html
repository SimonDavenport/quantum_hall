29 ////////////////////////////////////////////////////////////////////////////////
30 //! \brief Calculate the inner product between Term objects
31 //!
32 //! \return Numerical value of inner product
33 ////////////////////////////////////////////////////////////////////////////////
34 double InnerProduct(
35  const Term& bra, //!< Bra vector of inner product
36  const Term& ket) //!< Ket vector of inner product
37 {
38  // if the operators are the same (not including their coefficient)
39  // then we can exactly evaluate their inner product
40  // Make a copy of each vector, sort both to the same form
41  // and then check if they are the same or not (since the
42  // basis is orthogonal, if they differ then we get 0)
43  Term tempBra(bra);
44  Term tempKet(ket);
45  tempBra.SortVector();
46  tempKet.SortVector();
47  if(tempBra.SameTest(tempKet))
48  {
49  // Combine the bra and ket together
50  tempBra.Conjugate();
51  tempBra *= tempKet;
52  // Evaluate the operator exactly
53  tempBra.ExactResult();
54  return tempBra.GetCoefficient();
55  }
56  else
57  {
58  return 0.0;
59  }
60 }

187 ////////////////////////////////////////////////////////////////////////////////
188 //! \brief Calculates the commutator between the J_n and K_m U(1) current operators
189 //!
190 //! Here we are implementing the following commutation relations
191 //! - [J_n,J_m] = n delta_{m+n},0
192 //! - [K_n,K_m] = n delta_{m+n},0 ... etc.
193 //! - [J_n,K_m] = 0 ... etc.
194 //!
195 //! Also evaluates commutation relations for charged operators Y_k as
196 //!
197 //! - {Y_n,Y_m} = delta_{m+n},0
198 //! - [Y_n,J_m] = 0 etc.
199 //!
200 //! \return Value of the commutator
201 ////////////////////////////////////////////////////////////////////////////////
202 double Term::Commutator(
203  const char label1, //!< The colour label of the 1st U(1) operator (e.g. J, K etc.)
204  const int index1, //!< The U(1) index of the 1st U(1) operator (e.g. J_index1)
205  const char label2, //!< The colour label of the 2nd U(1) operator (e.g. J, K etc.)
206  const int index2) //!< The U(1) index of the 2nd U(1) operator (e.g. K_index2)
207  const
208 {
209  ////////////////////////////////////////////////////////////////////////////////
210  #if _BENCHMARK_MODE_==1
211  g_callsCommutator+=1;
212  #endif
213  ////////////////////////////////////////////////////////////////////////////////
214  if(label1==label2)
215  {
216  if(index1+index2==0)
217  {
218  if(_MAJORANA_CURRENT_ == label1)
219  {
220  return 1.0;
221  }
222  else
223  {
224  return (double)index1;
225  }
226  }
227  else
228  {
229  return 0.0;
230  }
231  }
232  else
233  {
234  return 0.0;
235  }

314 ////////////////////////////////////////////////////////////////////////////////
315 //! \brief Overloads the * operator such that objects of type Term 'multiply'
316 //! together in the usual way i.e. the coefficients multiply and the operator
317 //! list gets concatenated.
318 //!
319 //! We want "multiplication" to act such that for "this" * rhs, if "this" is an operator
320 //! and rhs is a ket then "this" picks up the m_occupations value from the rhs object and
321 //! then the result "lhs" is a ket vector.
322 //! If "this" is a bra and rhs is an operator then rhs should pick up the m_occupations
323 //! value of "this" and the result "lhs" is a bra vector.
324 //! If "this" and rhs are bra and ket then we shall define the result as orthogonal
325 //! unless their m_occupations values are the same.
326 //!
327 //! \return A new object of type Term which is a combination of the multiple objects.
328 ////////////////////////////////////////////////////////////////////////////////
329 Term Term::operator*=(
330  const Term& rhs) //!< Right-hand side of the expression as this = this * rhs
331 {
332  m_coefficient *= rhs.m_coefficient;
333  m_opArray.insert(m_opArray.end(), rhs.m_opArray.begin(), rhs.m_opArray.end());
334  m_opType.insert(m_opType.end(), rhs.m_opType.begin(), rhs.m_opType.end());
335  if(m_isOperator && !rhs.m_isOperator)
336  {
337  // If "this" is an operator and rhs is a ket then "this" picks up the
338  // m_occupations value from the rhs object and then the result "lhs" is
339  // a ket vector
340  m_occupations = rhs.m_occupations;
341  m_deltaNa0 = rhs.m_deltaNa0;
342  m_isOperator = false; // It's now a ket
343  }
344  else if(!m_isOperator && rhs.m_isOperator)
345  {
346  // If "this" is a bra and rhs is an operator then rhs should pick up the
347  // m_occupations value of "this" and the result "lhs" is a bra vector.
348  }
349  else if(!m_isOperator && !rhs.m_isOperator)
350  {
351  // If "this" and rhs are bra and ket then we shall define the result as
352  // orthogonal unless their m_occupations values are the same.
353  if(m_occupations != rhs.m_occupations)
354  {
355  m_coefficient=0.0;
356  }
357  }
358  else
359  {
360  // Complain if something confusing happened
361  if(m_occupations != rhs.m_occupations)
362  {
363  std::cerr << "ERROR with Term::operator* : Illegal operator combination!" << std::endl;
364  exit(EXIT_FAILURE);
365  }
366  }
367  return *this;
368 }

370 ////////////////////////////////////////////////////////////////////////////////
371 //! \brief Takes the right-most operator with positive index and commutes it to the
372 //! right. This function takes into account that we are evaluating a matrix element
373 //! with the vacuum state.
374 //!
375 //! This is done regardless of the operator label (e.g. J or K).
376 //! If the right-most operator is already on the far right, then do nothing
377 //! (it will be removed by a call to the 'ZeroTest' function)
378 //!
379 //! When the commutation takes place, it will potentially produce two terms.
380 //! The original uncommuted term will be overwritten with operators J_n and
381 //! J_m reversed. The additional term from the commutator [J_n,J_m] will be
382 //! appended to the list 'newTerm', as long as it is non-zero.
383 //!
384 //! Different commutation relations can be called depending on the U(1)
385 //! current label e.g. [J_n,k_m] can be different to [J_n,J_m] etc.
386 ////////////////////////////////////////////////////////////////////////////////
387 void Term::CommuteRight(
388  std::vector<Term>& newTerm) //!< A list where new terms can be added if they are non-zero
389 {
390  ////////////////////////////////////////////////////////////////////////////////
391  #if _BENCHMARK_MODE_==1
392  g_callsCommuteRight+=1;
393  #endif
394  ////////////////////////////////////////////////////////////////////////////////
395  int nbr = m_opArray.size();
396  // If the array is empty (just a constant) then return
397  if(nbr==0)
398  {
399  return;
400  }
401  // If the term is of "ascending form" and it contains an even number of operators
402  // then return the analytical result
403  if(((nbr&1)==0) && (IsAscendingForm()))
404  {
405  this->ExactResult();
406  return;
407  }
408  ////// Determine the right-most operator with positive index /////////////
409  int toCommute=nbr-1;
410  while(toCommute>0)
411  {
412  if(m_opArray[toCommute]>0)
413  {
414  break;
415  }
416  toCommute--;
417  }
418  ////// Do nothing if the right-most operator is already on the far right
419  if(toCommute==nbr-1)
420  {
421  return;
422  }
423  else // Commute it to the right using J_n J_m = J_m J_n + [J_n,J_m]
424  {
425  // Determine how many places we can commute the operator through "for free"
426  // exploiting the commutation relation being 0 when n+m !=0;
427  // From the position toCommute, search right until an operator is found which will give a non-zero
428  // commutation. Otherwise, move all the way to the right.
429  int commuteWith=toCommute+1;
430  while(m_opArray[commuteWith]+m_opArray[toCommute]!=0)
431  {
432  ++commuteWith;
433  if(commuteWith>=nbr)
434  {
435  commuteWith = nbr-1;
436  break;
437  }
438  }
439  // Generate a possible term given by the [J_n,J_m] part
440  double commutator = Commutator(m_opType[toCommute], m_opArray[toCommute], m_opType[commuteWith], m_opArray[commuteWith]);
441  if(commutator!=0)
442  {
443  Term returnTerm(*this); // make a copy of the original term
444  // remove the two U(1) current operators that we replaced by a commutator
445  returnTerm.m_opArray.erase(returnTerm.m_opArray.begin()+toCommute);
446  returnTerm.m_opType.erase(returnTerm.m_opType.begin()+toCommute);
447  returnTerm.m_opArray.erase(returnTerm.m_opArray.begin()+commuteWith-1);
448  returnTerm.m_opType.erase(returnTerm.m_opType.begin()+commuteWith-1);
449  // multiply the m_coefficient by the value of the commutator
450  returnTerm.m_coefficient *= commutator;
451  newTerm.push_back(returnTerm);
452  }
453  // Overwrite the original term swapping J_n and J_m:
454  // TODO implement - sign in the case of fermion modes
455  int tempInt = m_opArray[toCommute];
456  m_opArray[toCommute] = m_opArray[commuteWith];
457  m_opArray[commuteWith] = tempInt;
458  char tempChar = m_opType[toCommute];
459  m_opType[toCommute] = m_opType[commuteWith];
460  m_opType[commuteWith] = tempChar;
461  }
462 }

508 ////////////////////////////////////////////////////////////////////////////////
509 //! \brief A function to determine whether the current term will evaluate to zero
510 //! when taking a matrix element of the term with the vacuum.
511 //!
512 //! There are 5 possible cases:
513 //! - If there are an odd number of operators
514 //! - If the first term is such that we have <0| J_n<0 ...
515 //! - If the last term is such that we have ... J_n>0 |0>
516 //! - If the m_coefficient is zero (due to e.g. added terms cancelling)
517 //! - If there are no pairs of J_n J_-n (otherwise a single one would commute throught and annihilate the vacuum).
518 //!
519 //! \return If the term is zero then return true, otherwise false
520 ////////////////////////////////////////////////////////////////////////////////
521 bool Term::ZeroTest() const
522 {
523  ////////////////////////////////////////////////////////////////////////////////
524  #if _BENCHMARK_MODE_==1
525  g_callsZeroTest+=1;
526  #endif
527  ////////////////////////////////////////////////////////////////////////////////
528  const int nbrOps = m_opArray.size();
529  // the matrix element of the term will be zero if it contains an odd number of operators
530  if((nbrOps & 1)==1)
531  {
532  return true;
533  }
534  // The matrix element will be zero if it has a vanishing coefficient
535  const double zeroTol = 0.0000000000000001;
536  if(fabs(m_coefficient)<zeroTol)
537  {
538  return true;
539  }
540  // The matrix element will be zero if J_n>0 |0> or <0|J_n<0 occurs in it
541  if(nbrOps>0 && (m_opArray.front()<0 || m_opArray.back()>0) )
542  {
543  return true;
544  }
545  // Check for pairs of +n and -n
546  if(nbrOps>0)
547  {
548  std::vector<int> labelCounterPositive;
549  std::vector<int> labelCounterNegative;
550  std::vector<int> shiftIndex;
551  int nbrColours;
552  int maxIndex;
553  this->CountOperators(labelCounterPositive, labelCounterNegative, shiftIndex, &nbrColours, &maxIndex);
554  // Check to see if the number of J_n<0 and J_n>0 match up for each n
555  for(int i=0; i<nbrColours*maxIndex; ++i)
556  {
557  if(labelCounterPositive[i] != labelCounterNegative[i])
558  {
559  return true;
560  }
561  }
562  }
563  // otherwise it is non-zero
564  return false;
565 }

567 ////////////////////////////////////////////////////////////////////////////////
568 //! \brief A function to determine whether the current term will evaluate to zero
569 //! when evaluating a ket vector
570 //!
571 //! There are 3 possible cases:
572 //! - If the last term is such that we have ... J_n>0 |0>
573 //! - If the m_coefficient is zero (due to e.g. added terms cancelling)
574 //! - Not every operator J_n>0 can be paired with a J<n to its right
575 //!
576 //! \return If the term is zero then return true, otherwise false
577 ////////////////////////////////////////////////////////////////////////////////
578 bool Term::RightZeroTest() const
579 {
580  ////////////////////////////////////////////////////////////////////////////////
581  #if _BENCHMARK_MODE_==1
582  g_callsZeroTest+=1;
583  #endif
584  ////////////////////////////////////////////////////////////////////////////////
585  const int nbrOps=m_opArray.size();
586  // The matrix element will be zero if it has a vanishing coefficient
587  const double zeroTol = 0.0000000000000001;
588  if(fabs(m_coefficient)<zeroTol)
589  {
590  return true;
591  }
592  // The matrix element will be zero if J_n>0 |0> occurs in it
593  if(nbrOps>0 && m_opArray.back()>0 )
594  {
595  return true;
596  }
597  // Check that the number of J_n>0 is not greater than the number of J_n<0
598  // for a given n and colour
599  if(nbrOps>0)
600  {
601  std::vector<int> labelCounterPositive;
602  std::vector<int> labelCounterNegative;
603  std::vector<int> shiftIndex;
604  int nbrColours;
605  int maxIndex;
606  this->CountOperators(labelCounterPositive, labelCounterNegative, shiftIndex, &nbrColours, &maxIndex);
607  // Check to see if the number of J_n<0 is equal to or exceeds the number
608  // of J_n>0 for the same n and colour
609  for(int i=0; i<nbrColours*maxIndex; ++i)
610  {
611  if(labelCounterPositive[i]>labelCounterNegative[i])
612  {
613  return true;
614  }
615  }
616  }
617  // Additionally check that every J_n>0 there is a J_n<0 to its right,
618  // of the same colour
619  if(nbrOps>0)
620  {
621  for(int i=0; i<nbrOps; ++i)
622  {
623  short int value = m_opArray[i];
624  char type = m_opType[i];
625  if(value>0)
626  {
627  // Check if there is a corresponding n<0 term to its left
628  for(int j=i+1; j<nbrOps; ++j)
629  {
630  if(m_opType[j] == type && m_opArray[j]+value==0)
631  {
632  // it's non-zero
633  return false;
634  }
635  }
636  return false;
637  }
638  }
639  }
640  // otherwise it is non-zero
641  return false;
642 }

661 ////////////////////////////////////////////////////////////////////////////////
662 //! \brief Evaluates the number operators e.g. J_0 for each term.
663 //!
664 //! Since these operators commute with everything, they can simply
665 //! be replaced with the shifted number of particles in the cut of a given colour,
666 //! divided by the square root of the filling factor. NOTE: if we only consider
667 //! CF-EWF states, then the effective filling factor is nu=1 (a single LL of a
668 //! given colour).
669 //!
670 //! After doing this, we can remove the number operator(s) from the term.
671 //!
672 //! Note the number operator contains a factor of 1/sqrt(filling factor), however
673 //! that can be factored out, so it must be included in the original definition

719 //! \brief Test whether the term is in "ascending form" meaning that all
720 //! J_n<0 operators appear to the right of all J_n>0 operators
721 //!
722 //! If the term is in ascending form then a matrix element can be evaluated
723 //! using a simple analytical expression.
724 //!

727 bool Term::IsAscendingForm()
728 {
729  int nbrOps = m_opType.size();
730  if(nbrOps>0)
731  {
732  int nbrColours = 4;
733  std::vector<char> possibleColours(nbrColours);
734  possibleColours[0] = _U1_LABEL_1_;
735  possibleColours[1] = _U1_LABEL_2_;
736  possibleColours[2] = _U1_LABEL_3_;
737  possibleColours[3] = _MAJORANA_CURRENT_;
738  for(int c=0; c<nbrColours; ++c)
739  {
740  int signChanges = 0;
741  for(int i=0; i<nbrOps; ++i)
742  {
743  if(m_opType[i] == possibleColours[c])
744  {
745  if(m_opArray[i]>0 && signChanges == 0)
746  {
747  continue;
748  }
749  else if(m_opArray[i]<0 && signChanges == 1)
750  {
751  continue;
752  }
753  else if(m_opArray[i] == 0)
754  {
755  continue;
756  }
757  else if(signChanges>1)
758  {
759  return false;
760  }
761  else
762  {
763  signChanges+=1;
764  }
765  }
766  }
767  if(signChanges>1)
768  {
769  ++signChanges;
770  return false;
771  }
772  }
773  }
774  else
775  {
776  // Just a constant term on its own is in ascending form
777  return true;
778  }

782 ////////////////////////////////////////////////////////////////////////////////
783 //! \brief Evaluates the exact result for the case when the term is in "ascending
784 //! form" (see the function IsAscendingForm for the definition)
785 //!
786 //! The coefficient of this term is then simply given by
787 //! n1! (n2!) 2^n2 (n3!) 3^n3 ...
788 //! where n1 is the number of pairs of J_1 J_-1 etc. of each colour
789 //! (ignoring J_0 terms here)
790 //!
791 //! In addition, for the charged modes we get the factor
792 //!
793 //! n1! n2! n3! ...
794 //!
795 //! TODO implement correct sign in the case of majorana modes
796 ////////////////////////////////////////////////////////////////////////////////
797 void Term::ExactResult()
798 {
799  // We need to keep track of the counting of each type of
800  // label (colour, and index combined)
801  // To do that we need to know how many different types of
802  // these terms occur
803  std::vector<int> labelCounterPositive;
804  std::vector<int> labelCounterNegative;
805  std::vector<int> shiftIndex;
806  int nbrColours;
807  int maxIndex;
808  this->CountOperators(labelCounterPositive, labelCounterNegative, shiftIndex,
809  &nbrColours, &maxIndex);
810  // If the matrix element/ket does not contain an equal number of
811  // J_n>0 and J_n<0 of the same type, then exact evaluation is
812  // invalid
813  for(unsigned int i=0; i<labelCounterPositive.size(); ++i)
814  {
815  if(labelCounterPositive[i] != labelCounterNegative[i])
816  {
817  return;
818  }
819  }
820  // Now that we have the count, simply return the combined result
821  long int result = 1;
822  bool usedLabel1 = false;
823  bool usedLabel2 = false;
824  bool usedLabel3 = false;
825  bool usedLabel4 = false;
826  for(unsigned int i=0; i<m_opArray.size(); ++i)
827  {
828  if(m_opType[i]==_U1_LABEL_1_) usedLabel1 = true;
829  if(m_opType[i]==_U1_LABEL_2_) usedLabel2 = true;
830  if(m_opType[i]==_U1_LABEL_3_) usedLabel3 = true;
831  if(m_opType[i]==_MAJORANA_CURRENT_) usedLabel4 = true;
832  }
833  for(int i=0; i<maxIndex; ++i)
834  {
835  if(usedLabel1)
836  {
837  result *= utilities::Factorial<double>(labelCounterPositive[i])*
838  pow(i+1, labelCounterPositive[i]);
839  }
840  if(usedLabel2)
841  {
842  result *= utilities::Factorial<double>(labelCounterPositive[shiftIndex[0]+i])*
843  pow(i+1, labelCounterPositive[shiftIndex[0]+i]);
844  }
845  if(usedLabel3)
846  {
847  result *= utilities::Factorial<double>(labelCounterPositive[shiftIndex[1]+i])*
848  pow(i+1,labelCounterPositive[shiftIndex[1]+i]);
849  }
850  if(usedLabel4)
851  {
852  result *= utilities::Factorial<double>(labelCounterPositive[shiftIndex[2]+i]);
853  }
854  }
855  m_opArray.clear();
856  m_opType.clear();
857  m_coefficient *= result;
858  return;
859 }
