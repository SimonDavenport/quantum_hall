42  //!< Binomial look-up function protorype
43  //////////////////////////////////////////////////////////////////////////////////
44  //! \brief This function returns the combinatoric factor input1 Choose input2.
45  //! Inputs are double variables to avoid overflows.
46  //!
47  //! \return Value of binomial coefficient (output type T)
48  //////////////////////////////////////////////////////////////////////////////////
49  template <typename T>
50  T Binomial(
51  T input1, //!< First index of binomial
52  T input2) //!< Second index of binomial
53  {
54  if(input2>input1||input2<0){return 0;}
55  if(input2==0)
56  {
57  return 1;
58  }
59  T cmp = 1;
60  T output = 0;
61  if(input1==input2){return 1;}
62  if(input1>input2){output=input1;
63  while (input1-cmp>input2){output*=(input1-cmp);++cmp;}}
64  else if(input2>input1){output=input2;
65  while (input2-cmp>input1){output*=(input2-cmp);++cmp;}
66  output=1 / output;}
67  T in=input1-input2;
68  cmp=0;
69  while (in-cmp>1){output/=(in-cmp);++cmp;}
70  return (T)round(output);
71  }
