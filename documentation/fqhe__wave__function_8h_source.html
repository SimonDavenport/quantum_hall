283  std::string tempGeometry;

285 
286  #if _ENABLE_MPI_
287  if(0 == mpi.m_id) // For the master node
288  #endif
289  {
290  nbr = (*options)["nbr"].as<int>();
291  //nbrQh = (*options)["nbr-qh"].as<int>();
292  jastrowExponent = (*options)["jastrow-exponent"].as<int>();
293  nbrClusters = (*options)["cluster-size"].as<int>();
294 
295  if((*options)["type"].as<std::string>()=="l")
296  {
297  type = FQHE::_LAUGHLIN_;
298  }
299  else if((*options)["type"].as<std::string>()=="cf")

302  }
303  else if((*options)["type"].as<std::string>()=="bs")
304  {
305  type = FQHE::_BONDERSON_SLINGERLAND_;
306  }
307  else if((*options)["type"].as<std::string>()=="mr")
308  {
309  type = FQHE::_MOORE_READ_;
310  }
311  else if((*options)["type"].as<std::string>()=="nass")
312  {
313  type = FQHE::_NASS_;
314  }
315  else if((*options)["type"].as<std::string>()=="para")
316  {
317  type = FQHE::_PARAFERMION_;
318  }
319  else
320  {
321  std::cerr<<"\tERROR! unknown type "<<tempType<<std::endl;
322  exitFlag = true;
323  }
324 
325  if((*options)["bosonic"].as<bool>())
326  {
327  statistics = FQHE::_BOSONS_;
328  }
329  else
330  {
331  statistics = FQHE::_FERMIONS_;
332  }
333 
334  if((*options)["geometry"].as<std::string>()=="s")
335  {
336  geometry = FQHE::_SPHERE_;
337  }
338  else if((*options)["geometry"].as<std::string>()=="d")
339  {
340  geometry = FQHE::_DISC_;
341  }
342  else if((*options)["geometry"].as<std::string>()=="t")
343  {
344  geometry = FQHE::_TORUS_;
345  }
346  else
347  {
348  std::cerr<<"\tERROR! unknown geometry "<<tempGeometry<<std::endl;
349  exitFlag = true;
350  }
351 
352  ////// VARIABLES USED ONLY FOR LATTICE FQHE ////////////////////
353  #if _ENABLE_LATTICE_FQHE_
354 
355  Lx = (*options)["torus-x"].as<int>();
356  Ly = (*options)["torus-y"].as<int>();
357  torusState = (*options)["torus-state"].as<int>();
358 
359  #endif
360  ////////////////////////////////////////////////////////////////////////////
361  }
362 
363  #if _ENABLE_MPI_
364  mpi.Sync(&exitFlag,1,0);
365  #endif
366 
367  if(exitFlag)
368  {
369  exit(EXIT_FAILURE);
370  }
371 
372  #if _ENABLE_MPI_
373  this->MpiSync(0,mpi);
374  #endif
375 
376  return;
