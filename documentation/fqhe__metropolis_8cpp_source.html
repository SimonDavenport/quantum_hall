496 //////////////////////////////////////////////////////////////////////////////////
497 //! \brief This function performs a multi-particle random walk move on a around the
498 //! surface of a sphere.
499 //!
500 //! This function moves a random point a random distance around the surface of
501 //! The sphere. To move the point in an unbiased way, we must make the move in
502 //! the reference frame of the north pole:
503 //!
504 //! - Pick some point to move, with co-ordinates theta (denote it by "t_old")
505 //! and phi (denote it by "p_old") on a unit sphere.
506 //!
507 //! - We want to pick a random distance and direction to move the point in.
508 //! This is done by making a random 3D Cartesian vector dx with components
509 //! between -a and +a, where "a" is the maximum move distance. Then I make
510 //! dx+(0,0,1) and normalize it. This vector is now a unit vector pointing
511 //! to a point on the surface of the sphere almost at the north pole.
512 //! I think that there will be a uniform distribution of displacements of
513 //! that vector from (0,0,1).
514 //!
515 //! - Next we rotate this vector to the vicinity of the point that we want to
516 //! move (I will explain why this has to be done shortly). From the co-ordinates,
517 //! {t_old,p_old} we make a rotation matrix that will take the vector (0,0,1)
518 //! and map it to the direction {t_old,p_old}. The formula I use to get this
519 //! is given here: http://en.wikipedia.org/wiki/Rotation_matrix (scroll down
520 //! to where it says: "Rotation matrix from axis and angle").
521 //! Let's call this R(t_old,p_old). I then use R(t_old,p_old) to rotate the
522 //! random unit vector (dx+(0,0,1) normalized) to the location of the point
523 //! we wanted to move. This gives some new Cartesian vector (x,y,z) and then
524 //! from that I can get:
525 //!
526 //! p_new = arctan(y,x)
527 //! t_new = arccos(z)
528 //!
529 //! which are very close to the original p_old and t_old. The point is that
530 //! the value of {x,y,z} that one would get for the same dx would depend on
531 //! R(t_old,p_old). So you could not just take the original point {t_old,p_old} ,
532 //! add dx to it and normalize the result to unit length.
533 //////////////////////////////////////////////////////////////////////////////////
534 void FQHE::Metropolis::RandomMoveSphere()
535 {
536  for(unsigned int k=0; k<mcData->nbrToMove; ++k)
537  {
538  *(toMove+k) = mt.genrand_int31()%wfData->nbr;
539  for(unsigned int j=0; j<k; ++j)
540  {
541  if(*(toMove+k) == *(toMove+j))
542  k--;
543  }
544  }
545  double uvec[3]; // Cartesian unit vector perpendicular to axis
546  // of rotation: takes point to north pole
547  double rotn[3][3]; // Rotation matrix to take point to north pole
548  double dx[3]; // Randomly chosen Cartesian vector at north pole
549  double norm,sum; // Variables to generate a Cartesian unit vector
550  double newxyz[3]; // Values of the random Cartesian vector rotated
551  // back to the point to move
552  double cosTheta,sinTheta;// To store values sin(theta) and cos(theta)
553  double dist; // Distance to move the point
554  for(unsigned int k=0; k<mcData->nbrToMove; ++k)
555  {
556  // To change reference frame, we need to determine the rotation matrix taking the
557  // point to the north pole.
558  cosTheta = cos(*(theta+*(toMove+k)));
559  sinTheta = sin(*(theta+*(toMove+k)));
560  // Assign unit vector of rotation axis: (comes from the cross product of (x,y,z) with (0,0,r))
561  uvec[0] = -sin(*(phi+*(toMove+k)));
562  uvec[1] = cos(*(phi+*(toMove+k)));
563  // Calculate Cartesian rotation matrix:
564  rotn[0][0] = cosTheta+pow(uvec[0], 2.0)*(1-cosTheta);
565  rotn[0][1] = uvec[0]*uvec[1]*(1-cosTheta);
566  rotn[0][2] = uvec[1]*sinTheta;
567  rotn[1][0] = uvec[1]*uvec[0]*(1-cosTheta);
568  rotn[1][1] = cosTheta+pow(uvec[1], 2.0)*(1-cosTheta);
569  rotn[1][2] = -uvec[0]*sinTheta;
570  rotn[2][0] = -uvec[1]*sinTheta;
571  rotn[2][1] = uvec[0]*sinTheta;
572  rotn[2][2] = cosTheta;
573  // Pick a random Cartesian unit vector:
574  dx[0] = 2*mt.random()-1;
575  dx[1] = 2*mt.random()-1;
576  dx[2] = 2*mt.random()-1;
577  norm = sqrt(pow(dx[0], 2.0)+pow(dx[1], 2.0)+pow(dx[2], 2.0));
578  dist = mcData->maxDist*mt.random();
579  dx[0] = dist*(dx[0] /norm);
580  dx[1] = dist*(dx[1] /norm);
581  dx[2] = dist*(dx[2] /norm);
582  // Shift to north pole and renormalise
583  dx[2] = dx[2]+1.0;
584  norm = sqrt(pow(dx[0],2.0)+pow(dx[1],2.0)+pow(dx[2],2.0));
585  dx[0] = dx[0]/norm;
586  dx[1] = dx[1]/norm;
587  dx[2] = dx[2]/norm;
588  // Rotate back to vicinity of the point
589  for(int i=0; i<3; ++i)
590  {
591  sum = 0;
592  for(int j=0; j<3; ++j)
593  {
594  sum = sum+rotn[i][j]*dx[j];
595  }
596  newxyz[i] = sum;
597  }
598  // Convert to polar and spinor co-ordinates
599  *(phiNew+k) = atan2(newxyz[1],newxyz[0]);
600  *(thetaNew+k) = acos(newxyz[2]);
601  *(uNew+k) = dcmplx(sqrt((1.0+newxyz[2])/2)*cos(*(phiNew+k)/2), +sqrt((1.0+ newxyz[2])/2)*sin(*(phiNew+k)/2));
602  *(vNew+k) = dcmplx(sqrt((1.0-newxyz[2])/2)*cos(*(phiNew+k)/2), -sqrt((1.0-newxyz[2])/2)*sin(*(phiNew+k)/2));
603  // Store old values and update global values
604  *(vTmp+k) = *(v+*(toMove+k));
605  *(uTmp+k) = *(u+*(toMove+k));
606  *(u+*(toMove+k)) = *(uNew+k);
607  *(v+*(toMove+k)) = *(vNew+k);
608  }
609 }

766 //! (see O. Ciftja and C. Wexler PRB 67 075304 (2003))
767 //!
768 double FQHE::Metropolis::CoulombEnergy(int n, dcmplx *z)
769 {
770  double energy = (double)8*n/(3*PI*wfData->discRadius);
771  double dist;
772  for(int i=0; i<n; ++i)
773  {
774  dist = abs(*(z+i))/wfData->discRadius;
775  if(dist<5)
776  {

785  {
786  energy += 1.0/(n*abs(*(z+i)-*(z+j)));
787  }
788  }

900 //////////////////////////////////////////////////////////////////////////////////
901 //! \brief Function to check that the maximum occupancy condition is satisfied
902 //! for Metropolis moves on a lattice.
903 //!
904 //! \return true if the condition is satisfied and false if it is not
905 //////////////////////////////////////////////////////////////////////////////////
906 bool FQHE::Metropolis::CheckOccupancy(
907  std::complex<int>* config) //!< A lattice occupancy configuration
908 {
909  std::vector<std::complex<int> > list;
910  std::vector<std::complex<int> >::iterator k;
911  int counter = 1;
912  list.push_back(config[0]);
913  for(int i=1; i<wfData->nbr; ++i)
914  {
915  std::complex<int> element = config[i];
916  int flag = 0;
917  for(int j=0; j<counter; ++j)
918  {
919  if(real(element)==real(list[j]) && imag(element)==imag(list[j]))
920  {
921  flag = 1;
922  break;
923  }
924  }
925  if(flag==0)
926  {
927  list.push_back(element);
928  ++counter;
929  }
930  }
931  int occupations[counter];
932  for(int j=0; j<counter; ++j)
933  {
934  occupations[j] = 0;
935  }
936  for(int i=0; i<wfData->nbr; ++i)
937  {
938  std::complex<int> element = config[i];
939  for(int j=0; j<counter; ++j)
940  {
941  if(real(element)==real(list[j]) && imag(element)==imag(list[j]))
942  {
943  ++occupations[j];
944  }
945  }
946  }
947  for(int j=0; j<counter; ++j)
948  {
949  if(occupations[j]>wfData->maxOccupation)
950  {
951  return false;
952  }
953  }
954  return true;
955 }
