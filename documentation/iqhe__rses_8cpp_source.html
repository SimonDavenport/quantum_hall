62 //!
63 //! Calculate the monopole harmonic function, which describes the single
64 //! particle orbitals in the quantum Hall problem in a spherical geometry
65 //!
66 double IqheRses::EvaluateMonopoleHarmonic(
67  const double q, //!< monopole strength
68  const int l, //!< Landau level index
69  const double m, //!< z-component of angular momentum: from -q-l to q+l
70  const double x) //!< position to evaluate the monopole harmonic at
71  const
72 {
73  double sum = 0;
74  const int max = (int)(q+l-m);
75  for(int s=0; s<=max; ++s)
76  {
77  // Calculate -1^(q+l-m-s)
78  const int sign = (int)(q+l-m-s) & 1 ? -1 : 1;
79  sum += sign*utilities::BinomialFromTable(l, s)*utilities::BinomialFromTable((int)(2*q+l),
80  (int)(q+l-m-s))*std::pow((1-x), (l-s))*std::pow((1+x), s);
81  }
82  sum *= std::pow((1-x), ((q-m)/2.0))*std::pow((1+x), ((q+m)/2.0));
83  return sum;
84 }

86 //!
87 //! Calculate the normalization factor of the monopole harmonic functions
88 //!
89 double IqheRses::MonopoleHarmonicNorm(
90  const double q, //!< monopole strength
91  const int l, //!< Landau level index
92  const double m) //!< z-component of angular momentum: from -q-l to q+l
93  const
94 {
95  double norm = std::pow(2.0,(-q-l))*sqrt((2*(q+l)+1)/(4*PI));
96  if((q+l-m)>(q+l+m))
97  {
98  norm *= sqrt(utilities::DivFactorial<double>((q+l-m), (2.0*q+l))
99  *utilities::DivFactorial<double>((q+l+m),l));
100  }
101  else
102  {
103  norm *= sqrt(utilities::DivFactorial<double>((q+l-m), l)
104  *utilities::DivFactorial<double>((q+l+m),(2.0*q+l)));
105  }
106  return norm;
107 }

113  boost::program_options::variables_map* optionList, //!< Command line option list
114  const int lzaMin, //!< Minimum sector calculated
115  const int lzaMax) //!< Maximum sector calculated
116 {
117  utilities::cout.MainOutput() << "\n\t========================= CALCUALTING IQHE RSES "
118  << " ========================="<<std::endl;
119  int nbrParticles = (*optionList)["nbr-n"].as<int>();
120  int nbrA = (*optionList)["nbr-a"].as<int>();
121  if(nbrParticles>_MAX_NUMBER_)
122  {
123  std::cerr << "ERROR: IQHE spectrum not calculable for more than " << _MAX_NUMBER_
124  << " particles - the code needs to be modified." << std::endl;
125  return;
126  }
127  int nbrLevels = (*optionList)["nbr-ll"].as<int>();
128  double monopoleStrength = (double)(nbrParticles-nbrLevels*nbrLevels)/(2*nbrLevels);
129  // Dimension of the list of overlaps/pseudoenergies from -q-l to q+l with l=nbrLevels-1
130  int nbrStates = (int)(1+2*(monopoleStrength+nbrLevels-1));
131  utilities::cout.MainOutput() << "\n\t\tMONOPOLE STRENGTH (=FLUX/2)\t\t" << monopoleStrength << std::endl;
132  ////// Calculate list of overlap matrices
133  double* overlapMatrixBuffer = new double[nbrStates*nbrLevels*nbrLevels];
134  this->GenerateOverlapMatrix(overlapMatrixBuffer, monopoleStrength, nbrLevels, nbrStates);
135  double* energyLevels = new double[nbrStates*nbrLevels];
136  double normalization = 0.0;
137  this->GenerateNewLevels(overlapMatrixBuffer, energyLevels, normalization, nbrLevels, nbrStates);
138  delete[] overlapMatrixBuffer;
139  // Represent occupation patterns of single particle orbital populations by
140  // a fixed vector of 128-bit bitsets. Each bitset represents the occupation
141  // of single particle orbitals within one Landau level. The vector of these

143  for(int s=lzaMin; s<=lzaMax; ++s)
144  {
145  std::vector<std::vector<std::bitset<_MAX_NUMBER_>> > occupationsList;
146  std::vector<int> spectrumLabels;
147  occupationBasis::GenerateLevelOccupations(occupationsList, spectrumLabels,

149  utilities::cout.DebuggingInfo() << "\n\t\tOCCUPATION PATTERNS FOR SECTOR " << s << std::endl;
150  for(unsigned int i=0; i<occupationsList.size(); ++i)

152  utilities::cout.DebuggingInfo() << utilities::cout.HyphenLine() << std::endl;
153  for(int j=0; j<nbrLevels; ++j)
154  {
155  utilities::cout.DebuggingInfo() << "\t\t" << occupationsList[i][j] << std::endl;
156  }
157  }
158  #if _DEBUG_
159  this->CheckAngularMomentum(occupationsList, monopoleStrength, nbrA, nbrLevels, s);
160  #endif
161  // Allocate memory to store the RSES
162  double* spectrum = new double[occupationsList.size()];
163  occupationBasis::AssignEnergies(occupationsList, energyLevels, spectrum, normalization,
164  nbrStates, nbrLevels);

166  delete[] spectrum;
167  }
168  delete[] energyLevels;
169 }

172 //! \brief Generate the overlap matrix of monopole harmonic functions in the upper
173 //! half-sphere (assuming an equatorial real space cut) for functions with the
174 //! same z-component of angular momentum but occurring in different Landau levels.
175 //!
176 //! This calculation is necessary because such functions are not orthonormal when
177 //! integrated over only half of the sphere. The overlap matrix generated here
178 //! is diagonalized in order to generate a new set of single particle entanglement
179 //! energies for construction of the RSES
180 ////////////////////////////////////////////////////////////////////////////////
181 void IqheRses::GenerateOverlapMatrix(
182  double* overlapMatrixBuffer, //!< Buffer to store list of overlap matrices
183  const double q, //!< Monopole strength
184  const int nbrLevels, //!< Number of Landau levels
185  const int nbrStates) //!< Number of single particle orbitals
186  const
187 {
188  gsl_function integrand;
189  double integrandParams[4];
190  double lowerLimit = 0.0;
191  double upperLimit = 1.0;
192  double epsabs = 1e-10;
193  double epsrel = 1e-20;
194  double result;
195  double error;
196  size_t nevals;
197  size_t maxIntervals = 1000;
198  gsl_integration_cquad_workspace* gsl_work = gsl_integration_cquad_workspace_alloc(maxIntervals);
199  double* p_buffer = overlapMatrixBuffer;
200  double zComponentAngularMomentum = -q-(nbrLevels-1);
201  integrandParams[0] = q;
202  integrand.function = &Integrand;
203  for(int m=0;m<nbrStates; ++m, zComponentAngularMomentum += 1.0)
204  {
205  integrandParams[1] = zComponentAngularMomentum;
206  for(int i=0; i<nbrLevels; ++i)
207  {
208  integrandParams[2] = i;
209  const double iNorm = this->MonopoleHarmonicNorm(q, i, zComponentAngularMomentum);
210  for(int j=0; j<nbrLevels; ++j, ++p_buffer)
211  {
212  integrandParams[3] = j;
213  const double jNorm = this->MonopoleHarmonicNorm(q, j, zComponentAngularMomentum);
214  integrand.params = integrandParams;
215  gsl_integration_cquad(&integrand, lowerLimit, upperLimit, epsabs, epsrel, gsl_work,
216  &result, &error, &nevals);
217  *(p_buffer) = 2*PI*iNorm*jNorm*result;
218  }
219  }
220  }
221  gsl_integration_cquad_workspace_free(gsl_work);
222 }

224 //!
225 //! This function calls a LAPACK routine to diagonalize the overlap matrix
226 //! generated by the GenerateOverlapMatrix() function. This defines a new set of
227 //! entanglement energy levels
228 //!
229 void IqheRses::GenerateNewLevels(
230  double* overlapMatrixBuffer, //!< Buffer to store list of overlap matrices
231  double* energyLevels, //!< On input, contain alpha_m
232  //! On output contain epsilon_m
233  double& normalization, //!< A value for the normalization, to be set
234  const int nbrLevels, //!< Number of Landau levels
235  const int nbrStates) //!< Number of single particle orbitals
236  const
237 {
238  double* p_buffer = overlapMatrixBuffer;
239  double* p_levels = energyLevels;
240  for(int m=0; m<nbrStates; ++m)
241  {
242  utilities::linearAlgebra::DiagonalizeSymmetricMatrix<double>(p_buffer, p_levels, nbrLevels, 'U');
243  p_buffer += nbrLevels*nbrLevels;
244  p_levels += nbrLevels;
245  }
246  // Here we can exploit the fact that the spectrum is symmetrical
247  // Otherwise numerical errors can build up from the calculation of the
248  // overlap matrices
249  // Generate entanglement energy levels defined by the function log(e_n/(1-e_n))
250  // for a half-cut in real space (e_n are the eigenvalues of the overlap matrix)
251  // The normalization of the energy levels is defined by sum log(1-e_n)
252  // for all states.
253  normalization = 0.0;
254  utilities::cout.DebuggingInfo() << std::endl;
255  // Temporary store for the new energy levels
256  double* newLevels = new double[nbrStates*nbrLevels];
257  p_levels = newLevels;
258  for(int i=0; i<nbrStates*nbrLevels; ++i, ++p_levels)
259  {
260  const double alpha = energyLevels[i];
261  const double oneMinusAlpha = 1.0-alpha;
262  double currentNormalization = log(oneMinusAlpha);
263  *(p_levels) = log(alpha/oneMinusAlpha);
264  // inf or nan results can occur in the case where we calculated
265  // the overlap matrix at the end of the spectrum - these contain dummy states.
266  // OR they can occur due to numerical integration overflow
267  if(std::isinf(*(p_levels)) || std::isnan(*(p_levels)))
268  {
269  // in case of overflow error, set equal to - the value at the other
270  // end of the spectrum (assuming symmetry)
271  *(p_levels) = -newLevels[nbrStates*nbrLevels-i-1];
272  // if both ends are inf/nan then it corresponds to a dummy state
273  // e.g. in a lower Landau level at the end of the spectrum
274  // We can safely set the energy value to 0, and this will avoid propagation
275  // of inf and nan results (the corresponding state is never occupied)
276  if(std::isinf(*(p_levels)) || std::isnan(*(p_levels)))
277  {
278  *(p_levels) = 0;
279  }
280  }
281  if(std::isinf(currentNormalization) || std::isnan(currentNormalization))
282  {
283  // in case of overflow error, set equal to the value at the other
284  // end of the spectrum (assuming symmetry)
285  currentNormalization = log(energyLevels[nbrStates*nbrLevels-i-1]);
286  if(std::isinf(currentNormalization) || std::isnan(currentNormalization))
287  {
288  // If it's still inf or nan then we set the normalization contribution
289  // to be zero to avoid propagation of errors
290  currentNormalization = 0.0;
291  }
292  }
293  normalization += currentNormalization;
294  utilities::cout.DebuggingInfo() << "\tSingle particle pseudoenergy [" << i << "] = "
295  << *(p_levels) << std::endl;
296  }
297  memcpy(energyLevels, newLevels, nbrStates*nbrLevels*sizeof(double));
298  delete[] newLevels;
299  utilities::cout.DebuggingInfo() << "\n\tNormalization = " << normalization << std::endl;
300  return;
301 }
