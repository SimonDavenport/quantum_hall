33 template <typename Container>
34 void
35 stringtok (Container &container, string const &in,
36  const char * const delimiters = " \t\n")
37 {
38  const string::size_type len = in.length();
39  string::size_type i = 0;
40 
41  while ( i < len )
42  {
43  // eat leading whitespace
44  i = in.find_first_not_of (delimiters, i);
45  if (i == string::npos)
46  return; // nothing left but white space
47 
48  // find the end of the token
49  string::size_type j = in.find_first_of (delimiters, i);
50 
51  // push token
52  if (j == string::npos) {
53  container.push_back (in.substr(i));
54  return;
55  } else
56  container.push_back (in.substr(i, j-i));
57 
58  // set up for next loop
59  i = j + 1;
60  }
61 }
