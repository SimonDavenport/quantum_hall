156 //! \brief A template for a swap function, to swap the contents of two memory
157 //! addresses. This version is used as an alternative to the std::swap function
158 //! which does not quite have the correct behaviour for this application (since
159 //! std:swap swaps the pointers, not the addresses pointed to, given the same
160 //! set of arguments as with this Swap function).
161 //!
162 //////////////////////////////////////////////////////////////////////////////////

164 template <typename T>
165 void Swap(T*& a,T*& b)
166 {
167  const T temp = *a;
168  *a = *b;
169  *b = temp;
170 
171  return;
172 }

176 //////////////////////////////////////////////////////////////////////////////////
177 //! \brief Sorts keys and values lists according to the keys list. Returns
178 //! after the first nSort elements are sorted. Sorting is performed with
179 //! an in-place quick sort algorithm, with the pivot chosen as the median
180 //! of the first, middle and last element of the partition.
181 //!
182 //! Template arguments:
183 //! K: key data type, used for sorting
184 //! V: value data type, not used for sorting
185 //! O: Select from _ASCENDING_ORDER_ or _DESCENDING_ORDER_
186 //!
187 //////////////////////////////////////////////////////////////////////////////////
188 
189 template <typename K, typename V,order_t O>
190 void DoQuickSort(
191  K* keyList, //!< Address of sort key list
192  V* valueList, //!< Address of value list
193  const unsigned int& dim, //!< Dimension of list
194  const unsigned int& nSort, //!< Number of elements to sort before returning
195  unsigned int& maxSorted) //!< The value up to which the algorithm has now
196  //! sorted the input arrays
197 {
198  // An array of dimension 0 or 1 is sorted by definition
199 
200  if(dim==0 || dim==1) return;
201 
202  // Return if we've sorted a sufficient amount of the array already
203 
204  if(maxSorted>nSort) return;
205 
206  // Sort the first, middle and last values in order then set the pivot to
207  // the median of the first, middle and last values in the key list
208 
209  K* p_keyFirst = keyList;
210  K* p_keyMiddle = keyList+dim/2;
211  K* p_keyLast = keyList+dim-1;
212 
213  {
214  V* p_valueFirst = valueList;
215  V* p_valueMiddle = valueList+dim/2;
216  V* p_valueLast = valueList+dim-1;
217 
218  if(Compare<K,O>::Value(p_keyLast,p_keyMiddle))
219  {
220  Swap(p_keyLast,p_keyMiddle);
221  Swap(p_valueLast,p_valueMiddle);
222  }
223 
224  if(Compare<K,O>::Value(p_keyMiddle,p_keyFirst))
225  {
226  Swap(p_keyMiddle,p_keyFirst);
227  Swap(p_valueMiddle,p_valueFirst);
228  }
229 
230  if(Compare<K,O>::Value(p_keyLast,p_keyMiddle))
231  {
232  Swap(p_keyLast,p_keyMiddle);
233  Swap(p_valueLast,p_valueMiddle);
234  }
235  }
236 
237  // If the dimension is 2 or 3, by this stage it has been fully sorted
238 
239  if(dim==2 || dim==3)
240  {
241  // This value of max sorted is on average an underestimate to take
242  // into account the worst case scenario
243 
244  maxSorted += dim;
245 
246  return;
247  }
248 
249  // Set the pivot to the median value, which is now the middle value
250 
251  K pivot = *p_keyMiddle;
252 
253  // Starting with the pointers set to the far left and far right of the arrays,
254  // swap left and right values that are either side of the pivot value according
255  // to the comparison condition, then proceed by moving the pointers inwards
256  // until the pointers cross. The crossing point defines the partition of the
257  // array into sub-arrays. Note that the Compare function MUST return true
258  // if the compared values are equal
259 
260  K* p_keyLeft = keyList;
261  V* p_valueLeft = valueList;
262 
263  K* p_keyRight = keyList+dim-1;
264  V* p_valueRight = valueList+dim-1;
265 
266  unsigned int partitionSize = dim;
267 
268  while(p_keyRight > p_keyLeft)
269  {
270  while(Compare<K,O>::Value(p_keyLeft,&pivot) && p_keyLeft < p_keyLast)
271  {
272  p_keyLeft++;
273  p_valueLeft++;
274  }
275 
276  while(Compare<K,O>::Value(&pivot,p_keyRight) && p_keyRight > p_keyFirst)
277  {
278  p_keyRight--;
279  p_valueRight--;
280  partitionSize--;
281  }
282 
283  if(p_keyRight > p_keyLeft)
284  {
285  Swap(p_keyRight,p_keyLeft);
286  Swap(p_valueRight,p_valueLeft);
287  }
288  }
289 
290  // Recursively apply the sort algorithm to each sub-array
291 
292  DoQuickSort<K,V,O>(keyList,valueList,partitionSize,nSort,maxSorted);
293 
294  DoQuickSort<K,V,O>(keyList+partitionSize,valueList+partitionSize,dim-partitionSize,nSort,maxSorted);
295 
296  return;
297 };
