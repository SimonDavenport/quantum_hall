50  //! A class template to implement the Kahan summation algorithm for
51  //! objects of the high precision wrapper type

53  template<class T>
54  struct HighPrecKahanAccumulation
55  {
56  T m_sum;
57  T m_correction;
58  T m_tempSum;
59  T m_tempCorrection;
60  void Init()
61  {
62  m_sum.Set(0.0);
63  m_correction.Set(0.0);
64  }
65  T& GetTotal()
66  {
67  return m_sum;
68  }

71  ////////////////////////////////////////////////////////////////////////////////
72  //! \brief Defines a global template += operator overload for the Kahan
73  //! accumulator class. Call as simply a += b where a is an existing
74  //! HighPrecKahanAccumulation object and b is the numerical value to be added to
75  //! the sum (should be of the same type as the class type)
76  //!
77  //! \return address to the updated HighPrecKahanAccumulation object
78  ////////////////////////////////////////////////////////////////////////////////
79  template<typename T>
80  HighPrecKahanAccumulation<T>& operator+=(HighPrecKahanAccumulation<T>& a, const T& b)
81  {
82  a.m_tempCorrection = b;
83  a.m_tempSum = a.m_sum;
84  a.m_tempCorrection -= a.m_correction;
85  a.m_tempSum += a.m_tempCorrection;
86  a.m_correction = a.m_tempSum;
87  a.m_correction -= a.m_sum;
88  a.m_correction -= a.m_tempCorrection;
89  a.m_sum = a.m_tempSum;
90  return a;
91  }

93  ////////////////////////////////////////////////////////////////////////////////
94  //! \brief Defines a global template *= operator overload for the Kahan
95  //! accumulator class. Call as simply a *= b where a is an existing
96  //! HighPrecKahanAccumulation object and b is the numerical value to be multiplied
97  //! by (should be of the same type as the class type)
98  //!
99  //! \return address to the updated HighPrecKahanAccumulation object
100  ////////////////////////////////////////////////////////////////////////////////
101  template<typename T>
102  HighPrecKahanAccumulation<T>& operator*=(HighPrecKahanAccumulation<T>& a, const T& b)
103  {
104  a.m_sum *= b;
105  a.m_correction *= b;
106  return a;
107  }

110  //! \brief A generic template class HpWrap<complex type,int>
111  //!
112  //! This class is used to declare function overloads for +=, -=, *=, /= and =
113  //! (most of which are not declared for e.g. complex<double> variables)
114  //! it also defines multiplication of a complex number by an integer
115  //!
116  //! The following specializations are defined
117  //!
118  //! HpWrap<complex<double>,0> for complex doubles
119  //!
120  //! HpWrap<complex<__float128>,0> complex __float128 (from quadmath.h)
121  //!
122  //! HpWrap<mpc_t,prec> complex arbitary precision (mpc_t)

698  //////////////////////////////////////////////////////////////////////////////////
699  //! \brief Specialization of the cpWrap template class to complex<mpf_t> variables
700  //!
701  //! NOTE this is not a proper type, but it acts as a place holder to describe complex mpf variables
702  //! The type is implemented by an array containing two mpf_t types
703  //! Note that mpf_t typically a lot faster at arithmetic than mpfr_t or mpc_t,
704  //! However the gmp library does not contain special functions e.g log,
705  //! and it's rounding method does not confirm to IEEE.
706  //////////////////////////////////////////////////////////////////////////////////
707 
708  template <int P> class HpWrap<mpfCmplx,P>
709  {
710  private:
711  mpf_t m_x[2];
712  //!< m_x[0] is the real part and m_x[1] is the imaginary part
713 
714  public:
715 
716  HpWrap()
717  {
718  mpf_init2(m_x[0], P);
719  mpf_init2(m_x[1], P);
720  }
721 
722  ~HpWrap()
723  {
724  mpf_clear(m_x[0]);
725  mpf_clear(m_x[1]);
726  }
727 
728  //!
729  //! overload for the += operator
730  //!
731  HpWrap& operator+= (const HpWrap& rhs)
732  {
733  mpf_add(m_x[0], m_x[0], rhs.m_x[0]);
734  mpf_add(m_x[1], m_x[1], rhs.m_x[1]);
735  return *this;
736  }
737 
738  //!
739  //! overload for the -= operator
740  //!
741  HpWrap& operator-= (const HpWrap& rhs)
742  {
743  mpf_sub(m_x[0], m_x[0], rhs.m_x[0]);
744  mpf_sub(m_x[1], m_x[1], rhs.m_x[1]);
745  return *this;
746  }
747 
748  //!
749  //! overload for the *= operator
750  //!
751  HpWrap& operator*= (const HpWrap& rhs)
752  {
753  mpf_t tmp,tmp1;
754  mpf_init2(tmp, P);
755  mpf_init2(tmp1, P);
756  mpf_mul(tmp, m_x[0], rhs.m_x[0]);
757  mpf_mul(tmp1, m_x[1], rhs.m_x[1]);
758  mpf_sub(tmp, tmp, tmp1);
759  mpf_mul(m_x[1], m_x[1], rhs.m_x[0]);
760  mpf_mul(tmp1, m_x[0], rhs.m_x[1]);
761  mpf_add(m_x[1], m_x[1], tmp1);
762  mpf_set(m_x[0], tmp);
763  mpf_clear(tmp);
764  mpf_clear(tmp1);
765  return *this;
766  }

768  //!
769  //! overload for the /= operator
770  //!
771  HpWrap& operator/= (const HpWrap& rhs)
