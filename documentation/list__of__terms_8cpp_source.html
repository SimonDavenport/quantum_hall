29 //! \brief Calculate the inner product between a Term object and a Virasoro object
30 //!
31 //! \return Numerical value of inner product
32 ////////////////////////////////////////////////////////////////////////////////
33 double InnerProduct(
34  const Term& bra, //!< Bra vector of inner product
35  const ListOfTerms& ket) //!< Ket vector of inner product
36 {
37  // if the operators are the same (not including their coefficient)
38  // then we can exactly evaluate their inner product
39  const int size = ket.m_operator.size();
40  double innerProduct = 0.0;
41  const double zeroTol = 0.0000000000000001;
42  if(!fabs(ket.m_operator[0].GetCoefficient()) < zeroTol)
43  {
44  // Add terms to the inner product only if the ket is non-zero
45  for(int i=0; i<size; ++i)
46  {
47  innerProduct += InnerProduct(bra, (ket.m_operator[i]));
48  }
49  }
50  return innerProduct;
51 }

181 ////////////////////////////////////////////////////////////////////////////////
182 //! \brief A function to initiate the recursive commutation routine for ket
183 //! vectors.
184 //!
185 //! This will completely evaluate the numerical value of the matrix element
186 //! loaded into the object. Some short cuts are used for cases when the matrix
187 //! element is known to be zero based on the analytic properties of the
188 //! commutation relations e.g J_1 J_2 J_-1 |0> = 0 because J_2 commutes with
189 //! all the other terms. In fact, most matrix elements tend to be zero.
190 //!
191 ////////////////////////////////////////////////////////////////////////////////
192 void ListOfTerms::FullRightCommute()
193 {
194  // Evaluate e.g. J_0, K_0,... (number operator) terms first, since they commute with everything
195  this->EvaluateNumberOperators();
196  // First remove terms that evaluate to zero i.e. with J_n>0 |0> or <0| J_n<0
197  // or terms where there are no pairs of n and -n (if that's not the case then
198  // a J_n could be pulled through to annihilate the vacuum).
199  this->ClearRightZeroTerms();
200  bool doneTest = this->FullyCommutedTest();
201  while(!doneTest)
202  {
203  // For each term in the operator:
204  std::vector<Term> newTerms;
205  // Take the right-most operator with positive label and commute right
206  // if it is already on the far right, then evaluate the term to be zero
207  // Alternatively, convert the result to the form
208  // n1! (n2!) 2^n2 (n3!) 3^n3 ...
209  // where n1 is the number of pairs of J_1 J_-1 etc.
210  const int nbrTerms = m_operator.size();
211  for(int i=0; i<nbrTerms; ++i)
212  {
213  m_operator[i].CommuteRight(newTerms);
214  }
215  // Clear terms that are now zero
216  this->ClearRightZeroTerms();
217  // update the operator
218  m_operator.insert(m_operator.end(), newTerms.begin(), newTerms.end());
219  this->CombineSameTerms();
220  doneTest = this->FullyCommutedTest();
221  }
222  this->ClearRightZeroTerms();
223  this->SortAllVectors();
224  this->CombineSameTerms();
225  return;
226 }

228 ////////////////////////////////////////////////////////////////////////////////
229 //! \brief A function to determine whether each term will evaluate to zero or
230 //! not when taking a ket of the term with the vacuum.
231 //!
232 //! To do this, for each term we call ZeroTest from the Term class
233 //! (see term.cpp).
234 ////////////////////////////////////////////////////////////////////////////////
235 void ListOfTerms::ClearRightZeroTerms()
236 {
237  std::vector<Term> newTerms;
238  if(m_operator.size()>0)
239  {
240  for(unsigned int i=0; i<m_operator.size(); ++i)
241  {
242  // For each term check for
243  // 1. J_n |0> with n>0
244  // 3. 0 coefficient
245  // 4. Pairs of J_n J_-n
246  if(!m_operator[i].RightZeroTest())
247  {
248  newTerms.push_back(m_operator[i]);
249  }
250  }
251  }
252  // make sure that we have at least one constant term (even if it is zero)
253  if(newTerms.size()==0)
254  {
255  newTerms.push_back(Term(0.0,m_operator[0].m_occupations,m_operator[0].m_deltaNa0));
256  }
257  m_operator = newTerms;
258  return;
259 }

261 //!
262 //! A function to combine terms that have the same operator content.
263 //! We just generate a new term with the coefficients of the old like terms added.
264 //!
265 void ListOfTerms::CombineSameTerms()
266 {
267  std::vector<Term> newTerms;
268  std::vector<int> blackList;
269  const int nbrTerms=m_operator.size();
270  for(int i=0; i<nbrTerms; ++i)
271  {
272  int size=blackList.size();
273  int j=0;
274  bool proceed=true;
275  while(j<size)
276  {
277  if(blackList[j]==i)
278  {
279  proceed=false;
280  break;
281  }
282  ++j;
283  }
284  if(proceed)
285  {
286  double coefficient=m_operator[i].GetCoefficient();
287  for(int j=i+1; j<nbrTerms; ++j)
288  {
289  if(m_operator[i].SameTest(m_operator[j]))
290  {
291  blackList.push_back(j);
292  coefficient+=m_operator[j].GetCoefficient();
293  }
294  }
295  newTerms.push_back(m_operator[i]);
296  (newTerms.back()).SetCoefficient(coefficient);
297  }
298  }
299  m_operator = newTerms;
300  return;
301 }

345 //!
346 //! Test if the operator is fully right commuted i.e. it contains no
347 //! J_n>0 terms or is just a constant
348 //!
349 bool ListOfTerms::FullyCommutedTest() const
350 {
351  const unsigned int nbrTerms = m_operator.size();
352  // It's fully commuted if the only remaining term is a constant
353  if(nbrTerms==1)
354  {
355  if((m_operator[0].GetNbrU1())==0)
356  {
357  return true;
358  }
359  }
360  // It's fully commuted if every remaining term contains no J_n>0
361  if(nbrTerms>0)
362  {
363  for(unsigned int i=0; i<nbrTerms; ++i)
364  {
365  if(!m_operator[i].AllNegativeIndex())
366  {
367  return false;
368  }
369  }
370  return true;
371  }
372  // Otherwise it's not fully commuted
373  return false;
374 }
