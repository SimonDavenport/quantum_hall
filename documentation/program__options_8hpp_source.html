110  //!
111  //! Set parameters from command line argument list
112  //!
113  inline void InitFromCommandLine(
114  boost::program_options::variables_map* options, //!< Command line argument list
115  const utilities::MpiWrapper& mpi) //!< MPI wrapper class
116  {
117  if(0 == mpi.m_id) // For the master node
118  {
119  path = (*options)["path"].as<std::string>();
120  choice = (*options)["mode"].as<int>();
121  lzA = (*options)["lza"].as<int>();
122  realCut = (*options)["real-cut"].as<int>();
123  nbrFixed = (*options)["nbr-fixed"].as<int>();
124  nbrSect = (*options)["nbr-sect"].as<int>();
125  nbrFourier = (*options)["nbr-fourier"].as<int>();
126  rows = (*options)["rows"].as<int>();
127  columns = (*options)["columns"].as<int>();
128  renormFactor = (*options)["renorm"].as<int>();
129  }
130  // Synchronize all values with those set on the master node
131  this->MpiSync(0, mpi);
132  if(0 == choice)
133  {
134  exit(EXIT_SUCCESS);
135  }
136  return;
137  }
